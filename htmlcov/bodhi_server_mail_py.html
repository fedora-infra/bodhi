245def read_template(name: str) -> str: 

246 """ 

247 Read template text from file. 

248 

249 Args: 

250 name: The name of the email template stored in 'release' table in database. 

251 Returns: 

252 The text read from the file. 

253 """ 

254 location = config.get('mail.templates_basepath') 

255 directory = get_absolute_path(location) 

256 file_name = "%s.tpl" % (name) 

257 template_path = os.path.join(directory, file_name) 

258 

259 if os.path.exists(template_path): 

260 try: 

261 with open(template_path) as template_file: 

262 return template_file.read() 

263 except IOError as e: 

264 log.error("Unable to read template file: %s" % (template_path)) 

265 log.error("IO Error[%s]: %s" % (e.errno, e.strerror)) 

266 else: 

267 log.error("Path does not exist: %s" % (template_path))

270def get_template(update: 'Update', use_template: str = 'fedora_errata_template') -> list: 

271 """ 

272 Build the update notice for a given update. 

273 

274 Args: 

275 update: The update to generate a template about. 

276 use_template: The name of the variable in bodhi.server.mail that references the 

277 template to generate this notice with. 

278 Returns: 

279 A list of templates for the given update. 

280 """ 

281 from bodhi.server.models import UpdateStatus, UpdateType 

282 use_template = read_template(use_template) 

283 line = str('-' * 80) + '\n' 

284 templates = [] 

285 

286 for build in update.builds: 

287 h = get_rpm_header(build.nvr) 

288 info = {} 

289 info['date'] = str(update.date_pushed) 

290 info['name'] = h['name'] 

291 info['summary'] = h['summary'] 

292 info['version'] = h['version'] 

293 info['release'] = h['release'] 

294 info['url'] = h['url'] 

295 if update.status is UpdateStatus.testing: 

296 info['testing'] = ' Test' 

297 info['yum_repository'] = ' --enablerepo=updates-testing' 

298 else: 

299 info['testing'] = '' 

300 info['yum_repository'] = '' 

301 

302 info['subject'] = "%s%s%s Update: %s" % ( 

303 update.type is UpdateType.security and '[SECURITY] ' or '', 

304 update.release.long_name, info['testing'], build.nvr) 

305 info['updateid'] = update.alias 

306 info['description'] = h['description'] 

307 info['product'] = update.release.long_name 

308 info['notes'] = "" 

309 if update.notes and len(update.notes): 

310 info['notes'] = "Update Information:\n\n%s\n" % \ 

311 '\n'.join(wrap(update.notes, width=80)) 

312 info['notes'] += line 

313 

314 # Add this update's referenced Bugzillas 

315 i = 1 

316 info['references'] = "" 

317 if update.bugs: 

318 info['references'] = "References:\n\n" 

319 parent = True in [bug.parent for bug in update.bugs] 

320 for bug in update.bugs: 

321 # Don't show any tracker bugs for security updates 

322 if update.type is UpdateType.security: 

323 # If there is a parent bug, don't show trackers 

324 if parent and not bug.parent: 

325 log.debug("Skipping tracker bug %s" % bug) 

326 continue 

327 title = ( 

328 bug.title != 'Unable to fetch title' and bug.title != 'Invalid bug number') \ 

329 and ' - %s' % bug.title or '' 

330 info['references'] += " [ %d ] Bug #%d%s\n %s\n" % \ 

331 (i, bug.bug_id, title, bug.url) 

332 i += 1 

333 info['references'] += line 

334 

335 # generate a ChangeLog 

336 info['changelog'] = "" 

337 changelog = build.get_changelog(lastupdate=True) 

338 if changelog is not None: 338 ↛ 342line 338 didn't jump to line 342, because the condition on line 338 was never false

339 info['changelog'] = "ChangeLog:\n\n%s%s" % \ 

340 (changelog, line) 

341 

342 templates.append((info['subject'], use_template % info)) 

343 

344 return templates

347def _send_mail(from_addr: str, to_addr: str, body: str) -> None: 

348 """ 

349 Send emails with smtplib. This is a lower level function than send_e-mail(). 

350 

351 Args: 

352 from_addr: The e-mail address to use in the envelope from field. 

353 to_addr: The e-mail address to use in the envelope to field. 

354 body: The body of the e-mail. 

355 """ 

356 smtp_server = config.get('smtp_server') 

357 if not smtp_server: 

358 log.info('Not sending email: No smtp_server defined') 

359 return 

360 smtp = None 

361 try: 

362 log.debug('Connecting to %s', smtp_server) 

363 smtp = smtplib.SMTP(smtp_server) 

364 smtp.sendmail(from_addr, [to_addr], body.encode('utf-8')) 

365 except smtplib.SMTPRecipientsRefused as e: 

366 log.warning('"recipient refused" for %r, %r' % (to_addr, e)) 

367 except Exception: 

368 log.exception('Unable to send mail') 

369 finally: 

370 if smtp: 

371 smtp.quit()

374def send_mail(from_addr: str, to_addr: str, subject: str, body_text: str, 

375 headers: typing.Optional[dict] = None) -> None: 

376 """ 

377 Send an e-mail. 

378 

379 Args: 

380 from_addr: The address to use in the From: header. 

381 to_addr: The address to send the e-mail to. 

382 subject: The subject of the e-mail. 

383 body_text: The body of the e-mail to be sent. 

384 headers: A mapping of header fields to values to be included in the e-mail, 

385 if not None. 

386 """ 

387 if not from_addr: 

388 from_addr = config.get('bodhi_email') 

389 if not from_addr: 

390 log.warning('Unable to send mail: bodhi_email not defined in the config') 

391 return 

392 if to_addr in config.get('exclude_mail'): 

393 return 

394 

395 msg = [f'From: {from_addr}', f'To: {to_addr}'] 

396 if headers: 

397 for key, value in headers.items(): 

398 msg.append(f'{key}: {value}') 

399 msg.append(f"X-Bodhi: {config.get('default_email_domain')}") 

400 msg += [f'Subject: {subject}', '', body_text] 

401 body = '\r\n'.join(msg) 

402 

403 log.info('Sending mail to %s: %s', to_addr, subject) 

404 _send_mail(from_addr, to_addr, body)
