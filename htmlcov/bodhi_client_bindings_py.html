112def errorhandled(method: typing.Callable) -> typing.Callable: 

113 """Raise exceptions on failure. Used as a decorator for BodhiClient methods.""" 

114 @functools.wraps(method) 

115 def wrapper(*args, **kwargs) -> typing.Any: 

116 try: 

117 result = method(*args, **kwargs) 

118 # Bodhi allows comments to be written by unauthenticated users if they solve a Captcha. 

119 # Due to this, an authentication error is not raised by the server if the client fails 

120 # to authenticate for any reason, and instead an error about needing a captcha key is 

121 # presented instead. If we see that error, we can just raise an AuthError to trigger the 

122 # retry logic in the exception handler below. 

123 if 'errors' in result: 

124 for error in result['errors']: 

125 if 'name' in error and error['name'] == 'captcha_key': 

126 raise AuthError('Captcha key needed.') 

127 except AuthError: 

128 # An AuthError can be raised for four different reasons: 

129 # 

130 # 0) The password is wrong. 

131 # 1) The session cookies are expired. fedora.python does not handle this automatically. 

132 # 2) The session cookies are not expired, but are no longer valid (for example, this can 

133 # happen if the server's auth secret has changed.) 

134 # 3) The client received a captcha_key error, as described in the try block above. 

135 # 

136 # We don't know the difference between the cases here, but case #1 is fairly common and 

137 # we can work around it and case #2 by removing the session cookies and csrf token and 

138 # retrying the request. If the password is wrong, the second attempt will also fail but 

139 # we won't guard it and the AuthError will still be raised. 

140 args[0]._session.cookies.clear() 

141 args[0].csrf_token = None 

142 result = method(*args, **kwargs) 

143 

144 if 'errors' not in result: 

145 return result 

146 

147 # Otherwise, there was a problem... 

148 problems = 'An unhandled error occurred in the BodhiClient' 

149 try: 

150 problems = "\n".join([e['description'] for e in result['errors']]) 

151 except Exception: 

152 pass 

153 raise BodhiClientException(problems) 

154 return wrapper

157def _days_since(data_str: str) -> int: 

158 """ 

159 Return number of days since the datetime passed as input in the form '%Y-%m-%d %H:%M:%S'. 

160 

161 This can be used to calculate how many days an update is in current state by passing 

162 directly the 'date_pushed' or 'date_submitted' from the Update object. 

163 This is also useful to easily mock the output, since datetime.datetime.utcnow() 

164 cannot be mocked. 

165 

166 Args: 

167 data_str: The 'date_pushed' or 'date_submitted' from the Update object. 

168 

169 Returns: 

170 Number of days since the date in input. 

171 """ 

172 update_time = datetime.datetime.strptime(data_str, '%Y-%m-%d %H:%M:%S') 

173 return (datetime.datetime.utcnow() - update_time).days

356 @errorhandled 

357 def query(self, **kwargs) -> 'munch.Munch': 

358 """ 

359 Query bodhi for a list of updates. 

360 

361 Args: 

362 alias (str): The update alias. 

363 updateid (str): The update ID (eg: FEDORA-2015-0001). 

364 content_type (str): A content type (rpm, module) to limit the query to. 

365 releases (str): A comma separated list of releases that you wish to query updates 

366 for. 

367 status (str): The update status (``pending``, ``testing``, ``stable``, 

368 ``obsolete``, ``unpushed``) 

369 type (str): The type of the update: ``security``, ``bugfix``, 

370 ``enhancement``, and ``newpackage``. 

371 bugs (str): A comma separated list of Red Hat Bugzilla IDs. 

372 request (str): An update request to query for 

373 ``testing``, ``stable``, ``unpush``, ``obsolete`` or None. 

374 mine (bool): If True, only query the users updates. Default: False. 

375 packages (str): A space or comma delimited list of package names. 

376 limit (int): A deprecated argument, sets ``rows_per_page``. See its docstring for more 

377 info. 

378 approved_before (str): A datetime string. 

379 approved_since (str): A datetime string. 

380 builds (str): A space or comma delimited string of build nvrs. 

381 critpath (bool): A boolean to query only critical path updates. 

382 locked (bool): A boolean to filter only locked updates. 

383 modified_before (str): A datetime string to query updates that have been modified 

384 before a certain time. 

385 modified_since (str): A datetime string to query updates that have been modified 

386 since a certain time. 

387 pushed (bool): A boolean to filter only pushed updates. 

388 pushed_before (str): A datetime string to filter updates pushed before a certain 

389 time. 

390 pushed_since (str): A datetime string to filter updates pushed since a 

391 certain time. 

392 severity (str): A severity type to filter by (``unspecified``, 

393 ``urgent``, ``high``, ``medium``, ``low``). 

394 submitted_before (str): A datetime string to filter updates submitted 

395 before a certain time. 

396 submitted_since (str): A datetime string to filter updates submitted 

397 after a certain time. 

398 suggest (str): Query for updates that suggest a user restart 

399 (``logout``, ``reboot``). 

400 gating (str): filter by TestGatingStatus description. 

401 from_side_tag (bool): A boolean to filter updates created from side tag or from 

402 normal workflow. 

403 user (str): Query for updates submitted by a specific user. 

404 rows_per_page (int): Limit the results to a certain number of rows per page 

405 (min:1 max: 100 default: 20). 

406 page (int): Return a specific page of results. 

407 Returns: 

408 The response from Bodhi describing the query results. 

409 """ 

410 # bodhi1 compat 

411 if 'limit' in kwargs: 

412 kwargs['rows_per_page'] = kwargs['limit'] 

413 del(kwargs['limit']) 

414 # 'mine' may be in kwargs, but set False 

415 if kwargs.get('mine'): 

416 kwargs['user'] = self.username 

417 if 'package' in kwargs: 

418 # for Bodhi 1, 'package' could be a package name, build, or 

419 # update ID, so try and figure it out 

420 if re.search(UPDATE_TITLE_RE, kwargs['package']): 

421 kwargs['builds'] = kwargs['package'] 

422 elif re.search(UPDATE_ID_RE, kwargs['package']): 

423 kwargs['updateid'] = kwargs['package'] 

424 else: 

425 kwargs['packages'] = kwargs['package'] 

426 del(kwargs['package']) 

427 if 'release' in kwargs: 

428 if isinstance(kwargs['release'], list): 

429 kwargs['releases'] = kwargs['release'] 

430 else: 

431 kwargs['releases'] = [kwargs['release']] 

432 del(kwargs['release']) 

433 if 'type_' in kwargs: 

434 kwargs['type'] = kwargs['type_'] 

435 del(kwargs['type_']) 

436 # Old Bodhi CLI set bugs default to "", but new Bodhi API 

437 # checks for 'if bugs is not None', not 'if not bugs' 

438 if 'bugs' in kwargs and kwargs['bugs'] == '': 

439 kwargs['bugs'] = None 

440 return self.send_request('updates/', verb='GET', params=kwargs)

570 def init_username(self): 

571 """ 

572 Check to see if the username attribute on self is set, and set if if it is not. 

573 

574 If the username is already set on self, return. 

575 

576 If the username is not already set on self, attempt to find if there is a username that has 

577 successfully authenticated in the Fedora session file. If that doesn't work, fall back to 

578 prompting the terminal for a username. Once the username has been set, re-run 

579 self._load_cookies() so we can re-use the user's last session. 

580 """ 

581 if not self.username: 

582 if os.path.exists(fedora.client.openidbaseclient.b_SESSION_FILE): 

583 with open(fedora.client.openidbaseclient.b_SESSION_FILE) as session_cache: 

584 try: 

585 sc = json.loads(session_cache.read()) 

586 except ValueError: 

587 # If the session cache can't be decoded as JSON, it could be corrupt or 

588 # empty. Either way we can't use it, so let's just pretend it's empty. 

589 sc = {} 

590 for key in sc.keys(): 

591 if key.startswith(self.base_url) and sc[key]: 

592 self.username = key.split(f'{self.base_url}:')[1] 

593 break 

594 

595 if not self.username: 

596 self.username = input('Username: ') 

597 

598 self._load_cookies()

600 @errorhandled 

601 def csrf(self) -> str: 

602 """ 

603 Return CSRF token if already acquired, otherwise login, get a CSRF, cache it, and return. 

604 

605 If there is already a CSRF token, this method returns it. 

606 

607 If there is not, this method ensures that we know the username, logs in if we aren't already 

608 logged in acquires and caches a CSRF token, and returns it. 

609 

610 Returns: 

611 The CSRF token. 

612 """ 

613 if not self.csrf_token: 

614 self.init_username() 

615 if not self.has_cookies(): 

616 self.login(self.username, self.password) 

617 self.csrf_token = self.send_request( 

618 'csrf', verb='GET', auth=True)['csrf_token'] 

619 return self.csrf_token

623 Parse an update template file. 

624 

625 Args: 

626 input_file: The filename of the update template.
