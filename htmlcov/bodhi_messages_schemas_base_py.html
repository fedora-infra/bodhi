37class BodhiMessage(message.Message): 

38 """A base class for Bodhi messages.""" 

39 

40 @property 

41 def app_icon(self) -> str: 

42 """ 

43 Return a URL that points to the application's icon. 

44 

45 This is used when displaying the message to users. 

46 

47 Returns: 

48 A URL for Bodhi's app icon. 

49 """ 

50 return "https://apps.fedoraproject.org/img/icons/bodhi.png" 

51 

52 @property 

53 def agent(self) -> typing.Union[str, None]: 

54 """Return the agent's username for this message. 

55 

56 Returns: 

57 The agent's username, or None if the body has no agent key. 

58 """ 

59 return self.body.get('agent', None) 

60 

61 @property 

62 def agent_avatar(self) -> typing.Union[None, str]: 

63 """ 

64 Return a URL to the avatar of the user who caused the action. 

65 

66 Returns: 

67 The URL to the user's avatar, or None if username is None. 

68 """ 

69 username = self.agent 

70 if username is None: 

71 return None 

72 return user_avatar_url(username) 

73 

74 @classmethod 

75 def from_dict(cls, message: dict) -> 'BodhiMessage': 

76 """ 

77 Generate a message based on the given message dictionary. 

78 

79 Args: 

80 message: A dictionary representation of the message you wish to instantiate. 

81 Returns: 

82 A Message. 

83 """ 

84 # Dirty, nasty hack that I feel shame for: use the fedmsg encoder that modifies 

85 # messages quietly if they have objects with __json__ methods on them. 

86 # For now, copy that behavior. In the future, callers should pass 

87 # fedora_messaging.api.Message sub-classes or this whole API should go away. 

88 body = json.loads(json.dumps(message, cls=FedMsgEncoder)) 

89 

90 return cls(body=body) 

91 

92 @property 

93 def usernames(self) -> typing.List[str]: 

94 """ 

95 List of users affected by the action that generated this message. 

96 

97 Returns: 

98 A list of affected usernames. 

99 """ 

100 users = [] 

101 if self.agent is not None: 

102 users.append(self.agent) 

103 

104 if 'comment' in self.body: 

105 text = self.body['comment']['text'] 

106 mentions = re.findall(MENTION_RE, text) 

107 for mention in mentions: 

108 users.append(mention[1:]) 

109 

110 users = list(set(users)) 

111 users.sort() 

112 return users 

113

115 def containers(self) -> typing.Iterable[str]: 

116 """ 

117 List of containers affected by the action that generated this message. 

118 

119 Returns: 

120 A list of affected container names. 

121 """ 

122 return [] 

123 

124 @property 

125 def modules(self) -> typing.Iterable[str]: 

126 """ 

127 List of modules affected by the action that generated this message. 

128 

129 Returns: 

130 A list of affected module names. 

131 """ 

132 return [] 

133 

134 @property 

135 def flatpaks(self) -> typing.Iterable[str]: 

136 """ 

137 List of flatpaks affected by the action that generated this message. 

138 

139 Returns: 

140 A list of affected flatpaks names. 

141 """ 

142 return []

283class FedMsgEncoder(json.encoder.JSONEncoder): 

284 """Encoder with convenience support. 

285 

286 If an object has a ``__json__()`` method, use it to serialize to JSON. 

287 """ 

288 

289 def default(self, obj): 

290 """Encode objects which don't have a more specific encoding method.""" 

291 if hasattr(obj, "__json__"): 

292 return obj.__json__() 

293 return super().default(obj)
