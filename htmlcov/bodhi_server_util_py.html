68def get_rpm_header(nvr, tries=0): 

69 """ 

70 Get the rpm header for a given build. 

71 

72 Args: 

73 nvr (str): The name-version-release string of the build you want headers for. 

74 tries (int): The number of attempts that have been made to retrieve the nvr so far. Defaults 

75 to 0. 

76 Returns: 

77 dict: A dictionary mapping RPM header names to their values, as returned by the Koji client. 

78 Raises: 

79 ValueError: If no rpm headers found in koji. 

80 """ 

81 tries += 1 

82 headers = [ 

83 'name', 'summary', 'version', 'release', 'url', 'description', 

84 'changelogtime', 'changelogname', 'changelogtext', 

85 ] 

86 rpmID = nvr + '.src' 

87 koji_session = buildsys.get_session() 

88 try: 

89 result = koji_session.getRPMHeaders(rpmID=rpmID, headers=headers) 

90 except Exception as e: 

91 msg = "Failed %i times to get rpm header data from koji for %s: %s" 

92 log.warning(msg % (tries, nvr, str(e))) 

93 if tries < 3: 

94 # Try again... 

95 return get_rpm_header(nvr, tries=tries) 

96 else: 

97 # Give up for good and re-raise the failure... 

98 raise 

99 

100 if result: 

101 return result 

102 

103 raise ValueError("No rpm headers found in koji for %r" % nvr)

122class memoized(object): 

123 """Decorator that permanently caches a function's return value each time it is called. 

124 

125 If the function is called later with the same arguments, the cached value is returned (not 

126 reevaluated). 

127 

128 http://wiki.python.org/moin/PythonDecoratorLibrary#Memoize 

129 

130 Attributes: 

131 func (callable): The wrapped function. 

132 cache (dict): The cache, mapping arguments to the cached response. 

133 """ 

134 

135 def __init__(self, func): 

136 """ 

137 Initialize the memoized object. 

138 

139 Args: 

140 func: The function the memoized object is wrapping. 

141 """ 

142 self.func = func 

143 self.cache = {} 

144 

145 def __call__(self, *args): 

146 """ 

147 If the args are cached, return the cached value. If not, call the wrapped function. 

148 

149 If the wrapped function is called, it's response is only cached if the args are hashable. 

150 

151 Args: 

152 args (list): The list of arguments passed to the wrapped function. 

153 Returns: 

154 object: The response from the wrapped function, or the cached response, if available. 

155 """ 

156 try: 

157 hash(args) 

158 except TypeError: 

159 # uncacheable. a list, for instance. 

160 # better to not cache than blow up. 

161 return self.func(*args) 

162 if args in self.cache: 

163 return self.cache[args] 

164 else: 

165 value = self.func(*args) 

166 self.cache[args] = value 

167 return value 

168 

169 def __repr__(self): 

170 """ 

171 Return the function's docstring. 

172 

173 Returns: 

174 str: The wrapped function's docstring. 

175 """ 

176 return self.func.__doc__ 

177 

178 def __get__(self, obj, objtype): 

179 """ 

180 Support instance methods. 

181 

182 Args: 

183 obj (object): The instance of the object the wrapped method is bound to. 

184 objtype (type): The type of the instance of the object the wrapped method is bound to. 

185 Returns: 

186 callable: A functools.partial response with the wrapped method's instance passed to it. 

187 """ 

188 return functools.partial(self.__call__, obj)

227def sanity_check_repodata(myurl, repo_type): 

228 """ 

229 Sanity check the repodata for a given repository. 

230 

231 Args: 

232 myurl (str): A path to a repodata directory. 

233 repo_type (str): This should be set to 'yum' for Yum repositories, 'module' for module 

234 repositories, or 'source' for source repositories. 

235 Raises: 

236 RepodataException: If the repodata is not valid or does not exist. 

237 ValueError: If repo_type is not an acceptable value. 

238 """ 

239 if repo_type not in ('module', 'source', 'yum'): 

240 raise ValueError('repo_type must be one of module, source, or yum.') 

241 

242 with tempfile.TemporaryDirectory(prefix='bodhi_repotest_') as tmpdir: 

243 os.mkdir(os.path.join(tmpdir, 'lrodir')) 

244 

245 h = librepo.Handle() 

246 h.setopt(librepo.LRO_REPOTYPE, librepo.LR_YUMREPO) 

247 h.setopt(librepo.LRO_DESTDIR, os.path.join(tmpdir, 'lrodir')) 

248 

249 if myurl[-1] != '/': 

250 myurl += '/' 

251 if myurl.endswith('repodata/'): 

252 myurl = myurl.replace('repodata/', '') 

253 

254 h.setopt(librepo.LRO_URLS, [myurl]) 

255 h.setopt(librepo.LRO_LOCAL, True) 

256 h.setopt(librepo.LRO_CHECKSUM, True) 

257 h.setopt(librepo.LRO_IGNOREMISSING, False) 

258 r = librepo.Result() 

259 try: 

260 h.perform(r) 

261 except librepo.LibrepoException as e: 

262 rc, msg, general_msg = e.args 

263 raise RepodataException(msg) 

264 

265 repo_info = r.getinfo(librepo.LRR_YUM_REPO) 

266 reqparts = ['filelists', 'primary', 'repomd', 'updateinfo'] 

267 # Source and module repos don't have DRPMs. 

268 if repo_type == 'yum': 

269 reqparts.append('prestodelta') 

270 reqparts.append('group') 

271 elif repo_type == 'module': 

272 reqparts.append('modules') 

273 missing = [] 

274 for part in reqparts: 

275 if part not in repo_info: 

276 missing.append(part) 

277 if missing: 

278 raise RepodataException(f'Required parts not in repomd.xml: {", ".join(missing)}') 

279 

280 # Only yum repos have comps 

281 if repo_type == 'yum': 

282 # Test comps 

283 comps = libcomps.Comps() 

284 try: 

285 ret = comps.fromxml_f(repo_info['group']) 

286 except Exception: 

287 raise RepodataException('Comps file unable to be parsed') 

288 if len(comps.groups) < 1: 

289 raise RepodataException('Comps file empty') 

290 

291 # Test updateinfo 

292 ret = subprocess.call(['zgrep', '<id/>', repo_info['updateinfo']]) 

293 if not ret: 

294 raise RepodataException('updateinfo.xml.gz contains empty ID tags') 

295 

296 # Now call out to DNF to check if the repo is usable 

297 # "tests" is a list of tuples with (dnf args, expected output) to run. 

298 # For every test, DNF is run with the arguments, and if the expected output is not found, 

299 # an error is raised. 

300 tests = [] 

301 

302 if repo_type in ('yum', 'source'): 

303 tests.append((['list', 'available'], 'testrepo')) 

304 else: # repo_type == 'module', verified above 

305 tests.append((['module', 'list'], '.*')) 

306 

307 for test in tests: 

308 dnfargs, expout = test 

309 

310 # Make sure every DNF test runs in a new temp dir 

311 testdir = tempfile.mkdtemp(dir=tmpdir) 

312 output = sanity_check_repodata_dnf(testdir, myurl, *dnfargs) 

313 if (expout == ".*" and len(output.strip()) != 0) or (expout in output): 

314 continue 

315 else: 

316 raise RepodataException( 

317 "DNF did not return expected output when running test!" 

318 + f" Test: {dnfargs}, expected: {expout}, output: {output}")

321def sanity_check_repodata_dnf(tempdir, myurl, *dnf_args): 

322 """ 

323 Call DNF to try to parse and sanity check the repository. 

324 

325 Because DNF does not provide any usable API they won't break in a year time, we just shell out 

326 to run some repository actions on it. 

327 

328 Args: 

329 tempdir (str): Temporary directory that will be removed at the end. 

330 myurl (str): A path to a repodata directory. 

331 dnf_args (list): A list of arguments after DNF is set up to use the repo under test. 

332 Example would be ["search", "kernel"] or ["module", "list"]. 

333 Raises: 

334 Exception: If the repodata is not valid or does not exist. 

335 """ 

336 cmd = ['dnf', 

337 '--disablerepo=*', 

338 f'--repofrompath=testrepo,{myurl}', 

339 '--enablerepo=testrepo', 

340 '--setopt=skip_if_unavailable=0', 

341 '--setopt=testrepo.skip_if_unavailable=0', 

342 '--refresh', 

343 '--nogpgcheck'] + list(dnf_args) 

344 

345 return subprocess.check_output(cmd, encoding='utf-8')

373def avatar(context, username, size): 

374 """ 

375 Return a URL of an avatar for the given username of the given size. 

376 

377 Args: 

378 context (mako.runtime.Context): The current template rendering context. 

379 username (str): The username to return an avatar URL for. 

380 size (int): The size of the avatar you wish to retrieve, in unknown libravatar units. 

381 Returns: 

382 str: A URL to an avatar for the given username. 

383 """ 

384 # Handle some system users 

385 # https://github.com/fedora-infra/bodhi/issues/308 

386 if username in hardcoded_avatars: 

387 return hardcoded_avatars[username].format(size=size) 

388 

389 # context is a mako context object 

390 request = context['request'] 

391 https = request.registry.settings.get('libravatar_prefer_tls') 

392 

393 @request.cache.cache_on_arguments() 

394 def get_libravatar_url(openid, https, size): 

395 return libravatar.libravatar_url( 

396 openid=openid, 

397 https=https, 

398 size=size, 

399 default='retro', 

400 ) 

401 

402 def work(username, size): 

403 openid = "http://" + config.get('openid_template').format(username=username) + "/"

405 if config.get('libravatar_dns'): 

406 return get_libravatar_url(openid, https, size) 

407 else:

411 return template % (hash, query) 

412 

413 return 'libravatar.org' 

414 

415 return work(username, size)

418def splitter(value): 

419 """ 

420 Parse a string or list of comma or space delimited builds, returning a list of the values. 

421 

422 Examples: 

423 >>> util.splitter('one,two,,three,') 

424 ['one', 'two', 'three'] 

425 >>> util.splitter(['one,two,,three,,', 'four']) 

426 ['one', 'two', 'three', 'four'] 

427 

428 Args: 

429 value (str, colander.null, or iterable): The value to interpret as a list. 

430 Returns: 

431 list: A list of strings. 

432 """ 

433 if value == colander.null: 

434 return 

435 if isinstance(value, str): 

436 value = [value] 

437 

438 items = [] 

439 for v in value: 

440 if isinstance(v, str): 

441 for item in v.replace(',', ' ').split(): 

442 items.append(item) 

443 

444 elif v is not None: 

445 items.append(v) 

446 

447 return items

474def markup(context, text, bodhi=True): 

475 """ 

476 Return HTML from a markdown string. 

477 

478 Args: 

479 context (mako.runtime.Context): Unused. 

480 text (str): Markdown text to be converted to HTML. 

481 bodhi (bool): Enable or disable Bodhi markup extensions. 

482 Returns: 

483 str: HTML representation of the markdown text. 

484 """ 

485 markdown_attrs = { 

486 "img": ["src", "alt", "title"], 

487 "a": ["href", "alt", "title"], 

488 "div": ["class"], 

489 } 

490 markdown_tags = [ 

491 "h1", "h2", "h3", "h4", "h5", "h6", 

492 "b", "i", "strong", "em", "tt", 

493 "p", "br", 

494 "span", "div", "blockquote", "code", "hr", "pre", 

495 "ul", "ol", "li", "dd", "dt", 

496 "img", 

497 "a", 

498 ] 

499 

500 extensions = ['markdown.extensions.fenced_code', ] 

501 if bodhi == True: 

502 extensions.append(ffmarkdown.BodhiExtension()) 

503 markdown_text = markdown.markdown(text, extensions=extensions) 

504 

505 # previously, we linkified text in ffmarkdown.py, but this was causing issues like #1721 

506 # so now we use the bleach linkifier to do this for us. 

507 markdown_text = bleach.linkify(markdown_text, parse_email=True) 

508 

509 # previously, we used the Safe Mode in python-markdown to strip all HTML 

510 # tags. Safe Mode is deprecated, so we now use Bleach to sanitize all HTML 

511 # tags after running it through the markdown parser 

512 return bleach.clean(markdown_text, tags=markdown_tags, attributes=markdown_attrs)

568def pages_list(context, page, pages): 

569 """ 

570 Return a list of page numbers to display in pagination lists. 

571 

572 Args: 

573 context (mako.runtime.Context): Unused. 

574 page (int): The current page number. 

575 pages (int): The total number of pages. 

576 Returns: 

577 list: A list of page numbers that should be displayed to the user. 

578 """ 

579 margin = 4 

580 num_pages = (2 * margin) + 1 

581 

582 if page <= margin + 1: 

583 # Current `page` is close to the beginning of `pages` 

584 min_page = 1 

585 max_page = min(pages, num_pages) 

586 

587 elif (pages - page) >= margin: 

588 min_page = max(page - margin, 1) 

589 max_page = min(page + margin, pages) 

590 

591 else: 

592 # Current `page` is close to the end of `pages` 

593 max_page = min(pages, page + margin) 

594 min_page = max(max_page - (num_pages - 1), 1) 

595 

596 therange = list(range(min_page, max_page + 1)) 

597 if max_page != pages: 

598 therange = therange + ["...", pages] 

599 if min_page != 1: 

600 therange = [1, "..."] + therange 

601 

602 return therange

710def sorted_updates(updates): 

711 """ 

712 Sort the given iterable of Updates so the highest version appears last. 

713 

714 Order our updates so that the highest version gets tagged last so that 

715 it appears as the 'latest' in koji. 

716 

717 Args: 

718 updates (iterable): An iterable of bodhi.server.models.Update objects to be sorted. 

719 Returns: 

720 tuple: A 2-tuple of lists. The first list contains builds that should be tagged 

721 synchronously in a specific order. The second list can be tagged asynchronously in koji 

722 with a multicall. 

723 """ 

724 builds = defaultdict(set) 

725 sync, async_ = [], [] 

726 for update in updates: 

727 for build in update.builds: 

728 builds[build.nvr_name].add(build) 

729 # The sorted here is so we actually have a way to test this 

730 # Otherwise, we would be depending on the way Python orders dict keys 

731 for package in sorted(builds.keys()): 

732 if len(builds[package]) > 1: 

733 for build in sorted_builds(builds[package])[::-1]: 

734 if build.update not in sync: 

735 sync.append(build.update) 

736 if build.update in async_: 

737 async_.remove(build.update) 

738 else: 

739 build = list(builds[package])[0] 

740 if build.update not in async_ and build.update not in sync: 

741 async_.append(build.update) 

742 log.info('sync = %s', [up.alias for up in sync]) 

743 log.info('async_ = %s', [up.alias for up in async_]) 

744 if not (len(set(sync) & set(async_)) == 0 and len(set(sync) | set(async_)) == len(updates)): 

745 # There should be absolutely no way to hit this code path, but let's be paranoid, and check 

746 # every run, to make sure no update gets left behind. 

747 # It makes sure that there is no update in sync AND async, and that the combination of 

748 # sync OR async_ is the full set of updates. 

749 raise Exception('ERROR! SYNC+ASYNC != UPDATES! sorted_updates failed') # pragma: no cover 

750 return sync, async_

753def cmd(cmd, cwd=None, raise_on_error=False): 

754 """ 

755 Run the given command in a subprocess. 

756 

757 Args: 

758 cmd (list): The command to be run. This is expressed as a list to be 

759 passed directly to subprocess.Popen(). 

760 cwd (str or None): The current working directory to use when launching the 

761 subprocess. 

762 raise_on_error (bool): If True, raise a RuntimeError if the command's exit code is non-0. 

763 Defaults to False. 

764 Returns: 

765 tuple: A 3-tuple of the standard output (str), standard error (str), and the 

766 process's return code (int). 

767 Raises: 

768 RuntimeError: If exception is True and the command's exit code is non-0. 

769 """ 

770 log.debug('Running {}'.format(' '.join(cmd))) 

771 p = subprocess.Popen(cmd, cwd=cwd, shell=False, stdout=subprocess.PIPE, stderr=subprocess.PIPE) 

772 out, err = p.communicate() 

773 output = '{}\n{}'.format(out, err) 

774 if p.returncode != 0: 

775 msg = '{} returned a non-0 exit code: {}'.format(' '.join(cmd), p.returncode) 

776 log.error(msg) 

777 log.error(output) 

778 if raise_on_error: 

779 raise RuntimeError(msg) 

780 elif out or err: 

781 log.debug(f"subprocess output: {output}") 

782 return out, err, p.returncode

785def tokenize(string): 

786 """ 

787 Interpret the given string as a space or comma separated ordered iterable of strings. 

788 

789 For example, Given something like "a b, c d" return a generator equivalent of 

790 ['a', 'b', 'c', 'd']. 

791 

792 Args: 

793 string (str): The string to be interpreted as an iterable. 

794 Yields: 

795 str: The individual tokens found in the comma or space separated string. 

796 """ 

797 for substring in string.split(','): 

798 for token in substring.split(): 

799 # Using str.split() ensures that tokens don't contain white space 

800 # and aren't empty. Not specifying a separator "... means split 

801 # according to any whitespace, and discard empty strings from the 

802 # result." 

803 yield token

806def taskotron_results(settings, entity='results/latest', max_queries=10, **kwargs): 

807 """ 

808 Yield resultsdb results using query arguments. 

809 

810 Args: 

811 settings (bodhi.server.config.BodhiConfig): Bodhi's settings. 

812 entity (str): The API endpoint to use (see resultsdb documentation). 

813 max_queries (int): The maximum number of queries to perform (pages to retrieve). ``1`` means 

814 just a single page. ``None`` or ``0`` means no limit. Please note some tests might have 

815 thousands of results in the database and it's very reasonable to limit queries (thus the 

816 default value). 

817 kwargs (dict): Args that will be passed to resultsdb to specify what results to retrieve. 

818 Returns: 

819 generator or None: Yields Python objects loaded from ResultsDB's "data" field in its JSON 

820 response, or None if there was an Exception while performing the query. 

821 """ 

822 max_queries = max_queries or 0 

823 url = settings['resultsdb_api_url'] + "/api/v2.0/" + entity 

824 if kwargs: 

825 url = url + "?" + urlencode(kwargs) 

826 data = True 

827 queries = 0 

828 

829 try: 

830 while data and url: 

831 log.debug("Grabbing %r" % url) 

832 response = requests.get(url, timeout=60) 

833 if response.status_code != 200: 

834 raise IOError("status code was %r" % response.status_code) 

835 json = response.json() 

836 for datum in json['data']: 

837 yield datum 

838 

839 url = json.get('next') 

840 queries += 1 

841 if max_queries and queries >= max_queries and url: 

842 log.debug('Too many result pages, aborting at: %r' % url) 

843 break 

844 except Exception as e: 

845 log.exception("Problem talking to %r : %r" % (url, str(e)))

848class TransactionalSessionMaker(object): 

849 """Provide a transactional database scope around a series of operations.""" 

850 

851 @contextmanager 

852 def __call__(self): 

853 """ 

854 Manage a database Session object for the life of the context. 

855 

856 Yields a database Session object, then either commits the transaction if there were no 

857 Exceptions or rolls back the transaction. In either case, it also will close and remove the 

858 Session. 

859 """ 

860 session = Session() 

861 try: 

862 yield session 

863 session.commit() 

864 except Exception as e: 

865 # It is possible for session.rollback() to raise Exceptions, so we will wrap it in an 

866 # Exception handler as well so we can log the rollback failure and still raise the 

867 # original Exception. 

868 try: 

869 session.rollback() 

870 except Exception: 

871 log.exception('An Exception was raised while rolling back a transaction.') 

872 raise e 

873 finally: 

874 self._end_session() 

875 

876 def _end_session(self): 

877 """ 

878 Close and remove the session. 

879 

880 This has been split off the main __call__ method to make it easier to 

881 mock it out in unit tests. 

882 """ 

883 Session.remove()

940def get_critpath_components_from_pdc(branch, component_type='rpm', components=None): 

941 """ 

942 Search PDC for critical path packages based on the specified branch. 

943 

944 Args: 

945 branch (str): The branch name to search by. 

946 component_type (str): The component type to search by. Defaults to ``rpm``. 

947 components (frozenset or None): The list of components we are interested in. If None (the 

948 default), all components for the given branch and type are returned. 

949 Returns: 

950 list: Critical path package names. 

951 Raises: 

952 RuntimeError: If the PDC did not give us a 200 code. 

953 """ 

954 pdc_api_url = '{}/rest_api/v1/component-branches/'.format( 

955 config.get('pdc_url').rstrip('/')) 

956 query_args = { 

957 'active': 'true', 

958 'critical_path': 'true', 

959 'name': branch, 

960 'page_size': 100, 

961 'type': component_type, 

962 'fields': 'global_component' 

963 } 

964 # Create ordered dictionary with sorted query args to be able to compare URLs 

965 query_args = OrderedDict(sorted(query_args.items(), key=lambda x: x[0])) 

966 

967 critpath_pkgs_set = set() 

968 if components and len(components) < PDC_CRITPATH_COMPONENTS_GETALL_LIMIT: 

969 # Do a query for every single component 

970 for component in components: 

971 query_args['global_component'] = component 

972 pdc_api_url_with_args = '{0}?{1}'.format(pdc_api_url, urlencode(query_args)) 

973 pdc_request_json = pdc_api_get(pdc_api_url_with_args) 

974 for branch_rv in pdc_request_json['results']: 

975 critpath_pkgs_set.add(branch_rv['global_component']) 

976 if pdc_request_json['next']: 

977 raise Exception('We got paging when requesting a single component?!') 

978 else: 

979 pdc_api_url_with_args = '{0}?{1}'.format(pdc_api_url, urlencode(query_args)) 

980 while True: 

981 pdc_request_json = pdc_api_get(pdc_api_url_with_args) 

982 

983 for branch_rv in pdc_request_json['results']: 

984 critpath_pkgs_set.add(branch_rv['global_component']) 

985 

986 if pdc_request_json['next']: 986 ↛ 990line 986 didn't jump to line 990, because the condition on line 986 was never false

987 pdc_api_url_with_args = pdc_request_json['next'] 

988 else: 

989 # There are no more results to iterate through 

990 break 

991 return list(critpath_pkgs_set)

994def call_api(api_url, service_name, error_key=None, method='GET', data=None, headers=None, 

995 retries=0): 

996 """ 

997 Perform an HTTP request with response type and error handling. 

998 

999 Args: 

1000 api_url (str): The URL to query. 

1001 service_name (str): The service name being queried (used to form human friendly error 

1002 messages). 

1003 error_key (str): The key that indexes error messages in the JSON body for the given 

1004 service. If this is set to None, the JSON response will be used as the error message. 

1005 method (str): The HTTP method to use for the request. Defaults to ``GET``. 

1006 data (dict): Query string parameters that will be sent along with the request to the server. 

1007 headers (dict): The headers to send along with the request. 

1008 retries (int): The number of times to retry, each after a 1 second sleep, if we get a 

1009 non-200 HTTP code. Defaults to 3. 

1010 Returns: 

1011 dict: A dictionary representing the JSON response from the remote service. 

1012 Raises: 

1013 RuntimeError: If the server did not give us a 200 code. 

1014 """ 

1015 if data is None: 

1016 data = dict() 

1017 log.debug("Querying url: %s", api_url) 

1018 if method == 'POST': 

1019 if headers is None: 

1020 headers = {'Content-Type': 'application/json'} 

1021 base_error_msg = ( 

1022 'Bodhi failed to send POST request to {0} at the following URL ' 

1023 '"{1}". The status code was "{2}".') 

1024 rv = http_session.post(api_url, 

1025 headers=headers, 

1026 data=json.dumps(data), 

1027 timeout=60) 

1028 else: 

1029 base_error_msg = ( 

1030 'Bodhi failed to get a resource from {0} at the following URL ' 

1031 '"{1}". The status code was "{2}".') 

1032 rv = http_session.get(api_url, timeout=60) 

1033 

1034 if rv.status_code >= 200 and rv.status_code < 300: 

1035 return rv.json() 

1036 elif retries: 

1037 time.sleep(1) 

1038 return call_api(api_url, service_name, error_key, method, data, headers, retries - 1) 

1039 elif rv.status_code == 500: 

1040 log.debug(rv.text) 

1041 # There will be no JSON with an error message here 

1042 error_msg = base_error_msg.format( 

1043 service_name, api_url, rv.status_code) 

1044 log.error(error_msg) 

1045 raise RuntimeError(error_msg) 

1046 else: 

1047 log.debug(rv.text) 

1048 # If it's not a 500 error, we can assume that the API returned an error 

1049 # message in JSON that we can log 

1050 try: 

1051 rv_error = rv.json() 

1052 if error_key is not None: 

1053 rv_error = rv_error.get(error_key) 

1054 except ValueError: 

1055 rv_error = '' 

1056 error_msg = base_error_msg.format( 

1057 service_name, api_url, rv.status_code) 

1058 error_msg = '{0} The error was "{1}".'.format(error_msg, rv_error) 

1059 log.error(error_msg) 

1060 raise RuntimeError(error_msg)

1253def pyfile_to_module( 

1254 filename: str, modname: str, silent: bool = False) -> typing.Union[types.ModuleType, bool]: 

1255 """Create a Python module from a Python file. 

1256 

1257 This function behaves as if the file was imported as module. Copied from Flask's 

1258 ``flask.config.Config.from_pyfile`` method. 

1259 

1260 Args: 

1261 filename: the filename to load. This can either be an 

1262 absolute filename or a filename relative to the 

1263 current working directory. 

1264 modname: the name of the module that will be produced. 

1265 silent: set to ``True`` if you want silent failure for missing 

1266 files. 

1267 """ 

1268 filename = os.path.join(os.getcwd(), filename) 

1269 d = types.ModuleType(modname) 

1270 d.__file__ = filename 

1271 try: 

1272 with open(filename) as config_file: 

1273 exec(compile(config_file.read(), filename, 'exec'), d.__dict__)
