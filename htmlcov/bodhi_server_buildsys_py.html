41def multicall_enabled(func: typing.Callable[..., typing.Any]) -> typing.Callable[..., typing.Any]: 

42 """ 

43 Decorate the given callable to enable multicall handling. 

44 

45 This is used by DevBuildsys methods. 

46 

47 Args: 

48 func: The function to wrap. 

49 Returns: 

50 A wrapped version of func. 

51 """ 

52 @wraps(func) 

53 def wrapper(self, *args, **kwargs) -> typing.Any: 

54 """ 

55 If multicall is enabled, store the results from func on self. 

56 

57 If multicall is not enabled, just call func and return its results as per usual. 

58 """ 

59 if not self.multicall: 

60 return func(self, *args, **kwargs) 

61 

62 # disable multicall during execution, so that inner func calls to other 

63 # methods don't append their results as well 

64 self._multicall = False 

65 result = func(self, *args, **kwargs) 

66 self.multicall_result.append([result]) 

67 self._multicall = True 

68 return wrapper

717def wait_for_tasks( 

718 tasks: typing.List[typing.Any], 

719 session: typing.Union[koji.ClientSession, None] = None, 

720 sleep: int = 300) -> typing.List[typing.Any]: 

721 """ 

722 Wait for a list of koji tasks to complete. 

723 

724 Args: 

725 tasks: The return value of Koji's multiCall(). 

726 session: A Koji client session to use. If not provided, the 

727 function will acquire its own session. 

728 sleep: How long to sleep between polls on Koji when waiting for tasks to complete. 

729 Returns: 

730 A list of failed tasks. An empty list indicates that all tasks completed successfully. 

731 """ 

732 log.debug("Waiting for %d tasks to complete: %s" % (len(tasks), tasks)) 

733 failed_tasks = [] 

734 if not session: 

735 session = get_session() 

736 for task in tasks: 

737 if not task: 

738 log.debug("Skipping task: %s" % task) 

739 continue 

740 while not session.taskFinished(task): 

741 time.sleep(sleep) 

742 task_info = session.getTaskInfo(task) 

743 if task_info['state'] != koji.TASK_STATES['CLOSED']: 

744 log.error("Koji task %d failed" % task) 

745 failed_tasks.append(task) 

746 log.debug("%d tasks completed successfully, %d tasks failed." % ( 

747 len(tasks) - len(failed_tasks), len(failed_tasks))) 

748 return failed_tasks
