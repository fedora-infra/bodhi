69def postschema_validator(f): 

70 """ 

71 Modify a validator function, so that it is skipped if schema validation already failed. 

72 

73 Args: 

74 f (callable): The function we are wrapping. 

75 Returns: 

76 callable: The wrapped function. 

77 """ 

78 @wraps(f) 

79 def validator(request, **kwargs): 

80 """ 

81 Run the validator, but only if there aren't errors and there is validated data. 

82 

83 Args: 

84 request (pyramid.request.Request): The current web request. 

85 kwargs (dict): The other arguments to pass on to the wrapped validator. 

86 """ 

87 # The check on request.errors is to make sure we don't bypass other checks without 

88 # failing the request 

89 if len(request.validated) == 0 and len(request.errors) > 0: 

90 return 

91 

92 f(request, **kwargs) 

93 

94 return validator

115def cache_tags(request, build): 

116 """ 

117 Cache the tags for a koji build. 

118 

119 Args: 

120 request (pyramid.request.Request): The current request. 

121 build (str): The NVR of the build to cache. 

122 Returns: 

123 list or None: The list of tags, or None if there was a failure communicating with koji. 

124 """ 

125 if build in request.buildinfo and 'tags' in request.buildinfo[build]: 

126 return request.buildinfo[build]['tags'] 

127 tags = None 

128 try: 

129 tags = [tag['name'] for tag in request.koji.listTags(build)] 

130 if len(tags) == 0: 

131 request.errors.add('body', 'builds', 

132 'Cannot find any tags associated with build: %s' % build) 

133 except koji.GenericError: 

134 request.errors.add('body', 'builds', 

135 'Invalid koji build: %s' % build) 

136 # This might end up setting tags to None. That is expected, and indicates it failed. 

137 request.buildinfo[build]['tags'] = tags + request.from_tag_inherited 

138 return tags + request.from_tag_inherited

141def cache_release(request, build): 

142 """ 

143 Cache the builds release from the request. 

144 

145 Args: 

146 request (pyramid.request.Request): The current request. 

147 build (str): The NVR of the build to cache. 

148 Returns: 

149 Release or None: The release object, or None if no release can be matched to the tags 

150 associated with the build. 

151 """ 

152 if build in request.buildinfo and 'release' in request.buildinfo[build]: 

153 return request.buildinfo[build]['release'] 

154 tags = cache_tags(request, build) 

155 if tags is None: 

156 return None 

157 build_rel = Release.from_tags(tags, request.db) 

158 if not build_rel: 

159 msg = 'Cannot find release associated with ' + \ 

160 'build: {}, tags: {}'.format(build, tags) 

161 log.warning(msg) 

162 request.errors.add('body', 'builds', msg) 

163 # This might end up setting build_rel to None. That is expected, and indicates it failed. 

164 request.buildinfo[build]['release'] = build_rel 

165 return build_rel

168def cache_nvrs(request, build): 

169 """ 

170 Cache the NVR from the given build on the request, and the koji getBuild() response. 

171 

172 Args: 

173 request (pyramid.request.Request): The current request. 

174 build (str): The NVR of the build to cache. 

175 Raises: 

176 ValueError: If the build could not be found in koji. 

177 koji.GenericError: If an error was thrown by koji's getBuild() call. 

178 """ 

179 if build in request.buildinfo and 'nvr' in request.buildinfo[build]: 

180 return 

181 if build not in request.buildinfo: 

182 request.buildinfo[build] = {} 

183 

184 # Request info from koji, used to split NVR and determine type 

185 # We use Koji's information to get the NVR split, because modules can have dashes in their 

186 # stream. 

187 kbinfo = request.koji.getBuild(build) 

188 if not kbinfo: 

189 request.buildinfo[build]['info'] = None 

190 request.buildinfo[build]['nvr'] = None 

191 raise ValueError('Build %s did not exist' % build) 

192 request.buildinfo[build]['info'] = kbinfo 

193 request.buildinfo[build]['nvr'] = kbinfo['name'], kbinfo['version'], kbinfo['release']

386@postschema_validator 

387def validate_acls(request, **kwargs): 

388 """ 

389 Ensure the user has privs to create or edit an update. 

390 

391 There are two different code-paths that could pass through this validator: 

392 one of them is for submitting something new with a list of builds or a 

393 side-tag (a new update, or editing an update by changing the list of builds). 

394 The other is for changing the request on an existing update -- where an update 

395 alias has been passed to us. The latter is also used to check if a user has 

396 enough privs to display the edit form page in webUI. 

397 

398 We need to validate that the user has commit rights on all the build 

399 (either the explicitly listed ones or on the ones associated with the 

400 pre-existing update). If a provenpackager added a build for which the 

401 former update submitter doesn't have commit rights, the original submitter 

402 will lose their privs to edit the update. 

403 

404 In case of a side-tag update, we don't validate against the builds, 

405 but we require the submitter to be whom created the side-tag. 

406 The 'sidetag_owner' field is set by `validate_from_tag()`. 

407 

408 Args: 

409 request (pyramid.request.Request): The current request. 

410 kwargs (dict): The kwargs of the related service definition. Unused. 

411 """ 

412 if not request.user: 

413 # If you're not logged in, obviously you don't have ACLs. 

414 request.errors.add('cookies', 'user', 'No ACLs for anonymous user') 

415 return 

416 user = User.get(request.user.name) 

417 user_groups = [group.name for group in user.groups] 

418 acl_system = config.get('acl_system') 

419 

420 builds = None 

421 sidetag = None 

422 if 'builds' in request.validated: 

423 builds = request.validated['builds'] 

424 

425 if 'update' in request.validated: 

426 if request.validated['update'].release.state == ReleaseState.archived: 

427 request.errors.add('body', 'update', 'cannot edit Update for an archived Release') 

428 return 

429 builds = request.validated['update'].builds 

430 sidetag = request.validated['update'].from_tag 

431 

432 if not builds and not sidetag: 

433 log.warning("validate_acls was passed data with nothing to validate.") 

434 request.errors.add('body', 'builds', 'ACL validation mechanism was ' 

435 'unable to determine ACLs.') 

436 return 

437 

438 # Allow certain groups to push updates for any package 

439 admin_groups = config['admin_packager_groups'] 

440 for group in admin_groups: 

441 if group in user_groups: 

442 log.debug(f'{user.name} is in {group} admin group') 

443 return 

444 

445 # Make sure the user is in the mandatory packager groups. This is a 

446 # safeguard in the event a user has commit access on the ACL system 

447 # but isn't part of the mandatory groups. 

448 mandatory_groups = config['mandatory_packager_groups'] 

449 for mandatory_group in mandatory_groups: 

450 if mandatory_group not in user_groups: 

451 error = (f'{user.name} is not a member of "{mandatory_group}", which is a ' 

452 f'mandatory packager group') 

453 request.errors.add('body', 'builds', error) 

454 return 

455 

456 # If we try to create or edit a side-tag update, check if user owns the side-tag 

457 # The 'sidetag_owner' field is set by `validate_from_tag()`. 

458 if request.validated.get('from_tag') is not None: 

459 log.debug('Using side-tag validation method') 

460 sidetag = request.validated.get('from_tag') 

461 # the validate_from_tag() must have set the sidetag_owner field 

462 sidetag_owner = request.validated.get('sidetag_owner', None) 

463 if sidetag_owner is None: 

464 error = ('Update appear to be from side-tag, but we cannot determine ' 

465 'the side-tag owner') 

466 request.errors.add('body', 'builds', error) 

467 return 

468 

469 if sidetag_owner != user.name: 

470 request.errors.add('body', 

471 'builds', 

472 f'{user.name} does not own {sidetag} side-tag') 

473 request.errors.status = 403 

474 return 

475 elif 'update' in request.validated and sidetag: 

476 # This is a simplified check to avoid quering Koji for the side-tag owner 

477 # The user whom created the update is surely the one owning the side-tag 

478 update = request.validated['update'] 

479 if user == update.user: 

480 log.debug(f'{user.name} owns {update.alias} side-tag update') 

481 else: 

482 request.errors.add('body', 

483 'builds', 

484 f'{user.name} does not own {sidetag} side-tag') 

485 request.errors.status = 403 

486 return 

487 

488 # For normal updates, check against every build 

489 log.debug('Using builds validation method') 

490 for build in builds: 

491 # The whole point of the blocks inside this conditional is to determine 

492 # the "release" and "package" associated with the given build. For raw 

493 # (new) builds, we have to do that by hand. For builds that have been 

494 # previously associated with an update, we can just look it up no prob. 

495 if 'builds' in request.validated: 

496 # Split out NVR data unless its already done. 

497 cache_nvrs(request, build) 

498

500

502 try:

504 base=Package, build=buildinfo['info'])

510 request.errors.status = HTTPNotImplemented.code 

511 return 

512 

513 # Get the Package and Release objects 

514 package = Package.get_or_create(request.db, buildinfo) 

515 release = cache_release(request, build) 

516 if release is None: 

517 return 

518 elif 'update' in request.validated: 518 ↛ 528line 518 didn't jump to line 528, because the condition on line 518 was never false

519 buildinfo = request.buildinfo[build.nvr] 

520 

521 # Easy to find the release and package since they're associated 

522 # with a pre-stored Build obj. 

523 package = build.package 

524 release = build.update.release 

525 

526 # Now that we know the release and the package associated with this 

527 # build, we can ask our ACL system about it. 

528 has_access = False 

529 if acl_system == 'pagure': 

530 # Verify user's commit access 

531 try: 

532 has_access = package.hascommitaccess(user.name, release.branch) 

533 except RuntimeError as error:

541 # a generic error to the user 

542 log.exception(error)

544 'Please try again later.') 

545 request.errors.add('body', 'builds', error_msg) 

546 return 

547 people = [user.name] 

548 if has_access: 

549 # Retrieve people to be informed of the update 

550 try: 

551 people = package.get_pkg_committers_from_pagure()[0] 

552 except Exception: 

553 # This will simply mean no email will be posted to affected users 

554 # Just log it. 

555 log.warning(f'Unable to retrieve committers list from Pagure ' 

556 f'for {package.name}.') 

557 elif acl_system == 'dummy': 

558 committers = ['ralph', 'bowlofeggs', 'guest'] 

559 if config['acl_dummy_committer']: 

560 committers.append(config['acl_dummy_committer'])

563 people = committers 

564 else: 

565 log.warning('No acl_system configured') 

566 people = None 

567 

568 buildinfo['people'] = people 

569 

570 if not has_access: 

571 request.errors.add('body', 'builds', 

572 f'{user.name} does not have commit access to {package.name}') 

573 request.errors.status = 403

1129def _validate_override_build(request, nvr, db): 

1130 """ 

1131 Workhorse function for validate_override_builds. 

1132 

1133 Args: 

1134 request (pyramid.request.Request): The current request. 

1135 nvr (str): The NVR for a :class:`Build`. 

1136 db (sqlalchemy.orm.session.Session): A database session. 

1137 Returns: 

1138 bodhi.server.models.Build: A build that matches the given nvr. 

1139 """ 

1140 build = Build.get(nvr) 

1141 if build is not None: 

1142 if not request.validated['edited'] and \ 

1143 build.update is not None and \ 

1144 build.update.test_gating_status == TestGatingStatus.failed: 

1145 request.errors.add("body", "nvr", "Cannot create a buildroot override" 

1146 " if build's test gating status is failed.") 

1147 return 

1148 

1149 if not build.release: 

1150 # Oddly, the build has no associated release. Let's try to figure 

1151 # that out and apply it. 

1152 tag_types, tag_rels = Release.get_tags(request.db) 

1153 valid_tags = tag_types['candidate'] + tag_types['testing'] 

1154 

1155 tags = [tag['name'] for tag in request.koji.listTags(nvr) 

1156 if tag['name'] in valid_tags] 

1157 

1158 release = Release.from_tags(tags, db) 

1159 

1160 if release is None: 

1161 request.errors.add('body', 'nvr', 'Invalid build. Couldn\'t ' 

1162 'determine release from koji tags.') 

1163 return 

1164 

1165 build.release = release 

1166 

1167 if not build.release.override_tag: 

1168 request.errors.add("body", "nvr", "Cannot create a buildroot override because the" 

1169 " release associated with the build does not support it.") 

1170 return 

1171 

1172 for tag in build.get_tags(): 

1173 if tag in (build.release.candidate_tag, build.release.testing_tag): 

1174 # The build is tagged as a candidate or testing 

1175 break 

1176 

1177 else: 

1178 # The build is tagged neither as a candidate or testing, it can't 

1179 # be in a buildroot override 

1180 request.errors.add('body', 'nvr', 'Invalid build. It must be ' 

1181 'tagged as either candidate or testing.') 

1182 return 

1183 

1184 else: 

1185 tag_types, tag_rels = Release.get_tags(request.db) 

1186 valid_tags = tag_types['candidate'] + tag_types['testing'] 

1187 

1188 try: 

1189 tags = [tag['name'] for tag in request.koji.listTags(nvr) 

1190 if tag['name'] in valid_tags] 

1191 except Exception as e: 

1192 request.errors.add('body', 'nvr', "Couldn't determine koji tags " 

1193 "for %s, %r" % (nvr, str(e))) 

1194 return 

1195 

1196 release = Release.from_tags(tags, db) 

1197 

1198 if release is None: 

1199 request.errors.add('body', 'nvr', 'Invalid build') 

1200 return 

1201 

1202 build_info = request.koji.getBuild(nvr) 

1203 package = Package.get_or_create(db, 

1204 {'nvr': (build_info['name'], 

1205 build_info['version'], 

1206 build_info['release']), 

1207 'info': build_info}) 

1208 

1209 build_class = ContentType.infer_content_class( 

1210 base=Build, build=build_info) 

1211 build = build_class(nvr=nvr, release=release, package=package) 

1212 db.add(build) 

1213 db.flush() 

1214 

1215 return build
