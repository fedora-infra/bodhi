76 def _connect(self) -> None: 

77 """Create a Bugzilla client instance and store it on self._bz.""" 

78 user = config.get('bodhi_email') 

79 password = config.get('bodhi_password') 

80 url = config.get("bz_server") 

81 log.info("Using BZ URL %s" % url) 

82 if config['bugzilla_api_key']: 

83 self._bz = bugzilla.Bugzilla(url=url, api_key=config.get('bugzilla_api_key'), 

84 cookiefile=None, tokenfile=None) 

85 elif user and password: 

86 self._bz = bugzilla.Bugzilla(url=url, 

87 user=user, password=password, 

88 cookiefile=None, tokenfile=None) 

89 else: 

90 self._bz = bugzilla.Bugzilla(url=url, 

91 cookiefile=None, tokenfile=None)

116 def comment(self, bug_id: int, comment: str) -> None: 

117 """ 

118 Add a comment to the given bug. 

119 

120 Args: 

121 bug_id: The id of the bug you wish to comment on. 

122 comment: The comment to add to the bug. 

123 """ 

124 try: 

125 if len(comment) > 65535: 

126 raise InvalidComment(f"Comment is too long: {comment}") 

127 bug = self.bz.getbug(bug_id) 

128 attempts = 0 

129 while attempts < 5: 

130 try: 

131 bug.addcomment(comment) 

132 break 

133 except xmlrpc_client.Fault as e: 

134 attempts += 1 

135 log.error( 

136 f"\nA fault has occurred \nFault code: {e.faultCode}" 

137 f" \nFault string: {e.faultString}" 

138 ) 

139 except InvalidComment: 

140 log.error( 

141 "Comment too long for bug #%d: %s" % (bug_id, comment)) 

142 except xmlrpc_client.Fault as err: 

143 if err.faultCode == 102: 

144 log.info('Cannot retrieve private bug #%d.', bug_id) 

145 else: 

146 log.exception( 

147 "Got fault from Bugzilla on #%d: fault code: %d, fault string: %s", 

148 bug_id, err.faultCode, err.faultString) 

149 except Exception: 

150 log.exception("Unable to add comment to bug #%d" % bug_id)

152 def on_qa(self, bug_id: int, comment: str) -> None: 

153 """ 

154 Change the status of this bug to ON_QA if it is not already ON_QA, VERIFIED, or CLOSED. 

155 

156 This method will only operate on bugs that are associated with products listed 

157 in the bz_products setting. 

158 

159 This will also comment on the bug with some details on how to test and provide feedback for 

160 this update. 

161 

162 Args: 

163 bug_id: The bug id you wish to set to ON_QA. 

164 comment: The comment to be included with the state change. 

165 """ 

166 try: 

167 bug = self.bz.getbug(bug_id) 

168 if bug.product not in config.get('bz_products'): 

169 log.info("Skipping set on_qa on {0!r} bug #{1}".format(bug.product, bug_id)) 

170 return 

171 if bug.bug_status not in ('ON_QA', 'VERIFIED', 'CLOSED'): 

172 log.debug("Setting Bug #%d to ON_QA" % bug_id) 

173 bug.setstatus('ON_QA', comment=comment) 

174 else: 

175 bug.addcomment(comment) 

176 except xmlrpc_client.Fault as err: 

177 if err.faultCode == 102: 

178 log.info('Cannot retrieve private bug #%d.', bug_id) 

179 else: 

180 log.exception( 

181 "Got fault from Bugzilla on #%d: fault code: %d, fault string: %s", 

182 bug_id, err.faultCode, err.faultString) 

183 except Exception: 

184 log.exception("Unable to alter bug #%d" % bug_id)

186 def close(self, bug_id: int, versions: typing.Mapping[str, str], comment: str) -> None: 

187 """ 

188 Close the bug given by bug_id, mark it as fixed in the given versions, and add a comment. 

189 

190 This method will only operate on bugs that are associated with products listed 

191 in the bz_products setting. 

192 

193 Args: 

194 bug_id: The ID of the bug you wish to close. 

195 versions: A mapping of package names to nvrs of those packages that close the bug. 

196 comment: A comment to leave on the bug when closing it. 

197 """ 

198 args = {'comment': comment} 

199 try: 

200 bug = self.bz.getbug(bug_id) 

201 if bug.product not in config.get('bz_products'): 

202 log.info("Skipping set closed on {0!r} bug #{1}".format(bug.product, bug_id)) 

203 return 

204 # If this bug is for one of these builds... 

205 if bug.component in versions: 

206 version = versions[bug.component] 

207 # Get the existing list 

208 fixedin = [v.strip() for v in bug.fixed_in.split()] 

209 # Strip out any empty strings (already stripped) 

210 fixedin = [v for v in fixedin if v] 

211 

212 # There are Red Hat preferences to how this field should be 

213 # structured. We should use: 

214 # - the full NVR as it appears in koji 

215 # - space-separated if there's more than one. 

216 fixedin_str = " ".join(fixedin) 

217 

218 # Add our build if its not already there 

219 # but only if resultant string length is lower than 256 chars 

220 # See https://github.com/fedora-infra/bodhi/issues/1430 

221 if (version not in fixedin) and ((len(fixedin_str) + len(version)) < 255): 

222 args['fixedin'] = " ".join([fixedin_str, version]).strip() 

223 

224 bug.close('ERRATA', **args) 

225 except xmlrpc_client.Fault as err: 

226 if err.faultCode == 102: 

227 log.info('Cannot retrieve private bug #%d.', bug_id) 

228 else: 

229 log.exception( 

230 "Got fault from Bugzilla on #%d: fault code: %d, fault string: %s", 

231 bug_id, err.faultCode, err.faultString)

233 def update_details(self, bug: typing.Union['bugzilla.bug.Bug', None], 

234 bug_entity: 'models.Bug') -> None: 

235 """ 

236 Update the details on bug_entity to match what is found in Bugzilla. 

237 

238 Args: 

239 bug: The Bugzilla Bug we will use to update our own Bug object from. If None, 

240 bug_entity.bug_id will be used to fetch the object from Bugzilla. 

241 bug_entity: The bug we wish to update. 

242 """ 

243 if not bug: 

244 try: 

245 bug = self.bz.getbug(bug_entity.bug_id) 

246 except xmlrpc_client.Fault as err: 

247 if err.faultCode == 102: 

248 log.info('Cannot retrieve private bug #%d.', bug_entity.bug_id) 

249 bug_entity.title = 'Private bug' 

250 else: 

251 log.exception( 

252 "Got fault from Bugzilla on #%d: fault code: %d, fault string: %s", 

253 bug_entity.bug_id, err.faultCode, err.faultString) 

254 bug_entity.title = 'Invalid bug number' 

255 return 

256 except Exception: 

257 log.exception("Unknown exception from Bugzilla") 

258 return 

259 if bug.product == 'Security Response': 

260 bug_entity.parent = True 

261 bug_entity.title = bug.short_desc 

262 if isinstance(bug.keywords, str): 

263 keywords = bug.keywords.split() 

264 else: # python-bugzilla 0.8.0+ 

265 keywords = bug.keywords 

266 if 'security' in [keyword.lower() for keyword in keywords]: 

267 bug_entity.security = True
