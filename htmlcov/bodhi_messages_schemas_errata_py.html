16""" 

17Message schema for Errata. 

18 

19Each message is defined as a Python class. For details, see `fedora-messaging 

20<https://fedora-messaging.readthedocs.io/en/stable/>`_ documentation on 

21messages. 

22""" 

23 

24import typing 

25 

26from .base import BodhiMessage, BuildV1, ReleaseV1, SCHEMA_URL, UpdateV1, UserV1 

27 

28 

29class ErrataPublishV1(BodhiMessage): 

30 """Sent when an errata is published.""" 

31 

32 body_schema = { 

33 'id': f'{SCHEMA_URL}/v1/bodhi.errata.publish#', 

34 '$schema': 'http://json-schema.org/draft-04/schema#', 

35 'description': 'Schema for message sent when an update is pushed to stable', 

36 'type': 'object', 

37 'properties': { 

38 'body': { 

39 'type': 'string', 

40 'description': 'The body of an human readable message about the update', 

41 }, 

42 'subject': { 

43 'type': 'string', 

44 'description': 'A short summary of the update' 

45 }, 

46 'update': UpdateV1.schema(), 

47 }, 

48 'required': ['body', 'subject', 'update'], 

49 'definitions': { 

50 'build': BuildV1.schema(), 

51 } 

52 } 

53 

54 topic = "bodhi.errata.publish" 

55 

56 def __str__(self) -> str: 

57 """ 

58 Return a human-readable representation of this message. 

59 

60 This should provide a detailed representation of the message, much like the body 

61 of an email. 

62 """ 

63 return self.body['body'] 

64 

65 @property 

66 def agent(self) -> str: 

67 """Return the agent's username for this message. 

68 

69 Returns: 

70 The agent's username. 

71 """ 

72 return self.update.user.name 

73 

74 @property 

75 def packages(self) -> typing.Iterable[str]: 

76 """List of package names affected by the action that generated this message. 

77 

78 Returns: 

79 A list of affected package names. 

80 """ 

81 return self.update.packages 

82 

83 @property 

84 def summary(self) -> str: 

85 """ 

86 Return a short, human-readable representation of this message. 

87 

88 This should provide a short summary of the message, much like the subject line 

89 of an email. 

90 """ 

91 return self.body['subject'] 

92 

93 @property 

94 def url(self) -> str: 

95 """ 

96 Return a URL to the action that caused this message to be emitted. 

97 

98 Returns: 

99 A relevant URL. 

100 """ 

101 return f'https://bodhi.fedoraproject.org/updates/{self.update.alias}' 

102 

103 @property 

104 def update(self) -> UpdateV1: 

105 """Return the Update from this errata.""" 

106 if not hasattr(self, '_update'): 

107 # Let's cache the Update since a few different methods use it. 

108 self._update = UpdateV1( 

109 self.body['update']['alias'], 

110 [BuildV1(b['nvr']) for b in self.body['update']['builds']], 

111 UserV1(self.body['update']['user']['name']), self.body['update']['status'], 

112 self.body['update']['request'], ReleaseV1(self.body['update']['release']['name'])) 

113 return self._update
