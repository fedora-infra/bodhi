42@build.get(renderer='json', 

43 error_handler=bodhi.server.services.errors.json_handler) 

44def get_build(request): 

45 """ 

46 Retrieve a Build by name-version-release, specified via an "nvr" query string parameter. 

47 

48 Args: 

49 request (pyramid.request): The current web request. 

50 Returns: 

51 bodhi.server.models.Build or None: The Build matching the search, or None if there is no 

52 Build with the given NVR. 

53 """ 

54 nvr = request.matchdict.get('nvr') 

55 build = Build.get(nvr) 

56 if not build: 

57 request.errors.add('body', 'nvr', 'No such build') 

58 request.errors.status = HTTPNotFound.code 

59 return 

60 return build

63@builds.get(schema=bodhi.server.schemas.ListBuildSchema, renderer='json', 

64 error_handler=bodhi.server.services.errors.json_handler, 

65 validators=(colander_querystring_validator, validate_releases, validate_updates, 

66 validate_packages)) 

67def query_builds(request): 

68 """ 

69 Search for Builds by given criteria. 

70 

71 The following criteria can be provided via query string parameters to search for Builds: 

72 nvr: The dash-separated name-version-release of a Build. 

73 updates: A space or comma separated list of updates to limit the search by. 

74 packages: A space or comma separated list of packages to search for builds by. 

75 releases: A space or comma separated list of release ids to limit builds by. 

76 page: Which page of search results are desired. 

77 rows_per_pags: How many results per page are desired. 

78 

79 Args: 

80 request (pyramid.request): The current request, containing the search criteria documented 

81 above. 

82 Returns: 

83 dict: A dictionary with the following key value mappings: 

84 builds: An iterable of builds that match the search criteria, ordered by nvr. 

85 page: The current page. 

86 pages: The total number of pages. 

87 rows_per_page: The number of rows per page. 

88 total: The number of builds that match the search criteria. 

89 """ 

90 db = request.db 

91 data = request.validated 

92 query = db.query(Build).order_by(Build.nvr.asc()) 

93 

94 nvr = data.get('nvr') 

95 if nvr is not None: 

96 query = query.filter(Build.nvr == nvr) 

97 

98 updates = data.get('updates') 

99 if updates is not None: 

100 query = query.join(Build.update) 

101 args = [Update.alias == update.alias for update in updates] 

102 query = query.filter(or_(*args)) 

103 

104 packages = data.get('packages') 

105 if packages is not None: 

106 query = query.join(Build.package) 

107 query = query.filter(or_(*[Package.id == p.id for p in packages])) 

108 

109 releases = data.get('releases') 

110 if releases is not None: 

111 query = query.join(Build.release) 

112 query = query.filter(or_(*[Release.id == r.id for r in releases])) 

113 

114 # We can't use ``query.count()`` here because it is naive with respect to 

115 # all the joins that we're doing above. 

116 count_query = query.with_labels().statement\ 

117 .with_only_columns([func.count(distinct(Build.nvr))])\ 

118 .order_by(None) 

119 total = db.execute(count_query).scalar() 

120 

121 page = data.get('page') 

122 rows_per_page = data.get('rows_per_page') 

123 pages = int(math.ceil(total / float(rows_per_page))) 

124 query = query.offset(rows_per_page * (page - 1)).limit(rows_per_page) 

125 

126 return dict( 

127 builds=query.all(), 

128 page=page, 

129 pages=pages, 

130 rows_per_page=rows_per_page, 

131 total=total, 

132 )
