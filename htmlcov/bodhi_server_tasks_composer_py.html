60def checkpoint(method): 

61 """ 

62 Decorate a method for skipping sections of the compose when resuming. 

63 

64 Args: 

65 method (callable): The callable to skip if we are resuming. 

66 Returns: 

67 callable: A function that skips the method if it can. 

68 """ 

69 key = method.__name__ 

70 

71 @functools.wraps(method) 

72 def wrapper(self, *args, **kwargs): 

73 if not self.resume or not self._checkpoints.get(key): 

74 # Call it 

75 retval = method(self, *args, **kwargs) 

76 if retval is not None: 

77 raise ValueError("checkpointed functions may not return stuff") 

78 # if it didn't raise an exception, mark the checkpoint 

79 self._checkpoints[key] = True 

80 self.save_state() 

81 else: 

82 # cool! we don't need to do anything, since we ran last time 

83 pass 

84 

85 return None 

86 return wrapper

89class ComposerHandler(object): 

90 """ 

91 The Bodhi Composer. 

92 

93 A consumer that listens for messages from releng members. 

94 

95 An updates "compose" consists of: 

96 

97 - Verify that the message was sent by someone in releng 

98 - Determine which updates to push 

99 - Lock repo 

100 - track which repos were completed 

101 - track which packages are in the push 

102 - lock updates 

103 - Make sure things are safe to move? (ideally we should trust our own state) 

104 - Check with taskotron to see if updates are pushable. 

105 - Update security bug titles 

106 - Move build tags 

107 - Expire buildroot overrides 

108 - Remove pending tags 

109 - Send messages 

110 - compose 

111 

112 Things to do while we're waiting on compose: 

113 

114 - Add testing updates to updates-testing digest 

115 - Generate/update updateinfo.xml 

116 - Have a coffee. Have 7 coffees. 

117 

118 Once compose is done: 

119 

120 - inject the updateinfo it into the repodata 

121 - Sanity check the repo 

122 - Flip the symlinks to the new repo 

123 - Cache the new repodata 

124 

125 - Generate and email stable update notices 

126 - Wait for the repo to hit the master mirror 

127 - Update bugzillas 

128 - Add comments to updates 

129 - Email updates-testing digest 

130 - request_complete 

131 

132 - Unlock repo 

133 - unlock updates 

134 - see if any updates now meet the stable criteria, and set the request 

135 """

168 def run(self, api_version: int, data: dict): 

169 """ 

170 Begin the push process. 

171 

172 Here we organize & prioritize the updates, and fire off separate 

173 threads for each repo tag being composed. 

174 

175 If there are any security updates in the push, then those repositories 

176 will be executed before all others. 

177 

178 Args: 

179 api_version: API version number. 

180 data: Information about the compose job we are processing. 

181 """ 

182 resume = data.get('resume', False) 

183 agent = data.get('agent') 

184 notifications.publish( 

185 compose_schemas.ComposeStartV1.from_dict(dict(agent=agent)), 

186 force=True) 

187 

188 results = [] 

189 threads = [] 

190 for compose in self._get_composes(api_version, data): 

191 log.info('Now starting composes') 

192 

193 composer = get_composer(ContentType.from_string(compose['content_type'])) 

194 if not composer: 

195 log.error( 

196 'Unsupported content type %s submitted for composing. SKIPPING', 

197 compose['content_type'] 

198 ) 

199 continue 

200 

201 thread = composer(self.max_composes_sem, compose, agent, self.db_factory, 

202 self.compose_dir, resume) 

203 threads.append(thread) 

204 thread.start() 

205 

206 log.info('All of the batches are running. Now waiting for the final results') 

207 for thread in threads: 

208 thread.join() 

209 for result in thread.results(): 

210 results.append(result) 

211 

212 log.info('Push complete! Summary follows:') 

213 for result in results: 

214 log.info(result)

216 def _get_composes(self, api_version: int, data: dict): 

217 """ 

218 Return a list of dictionaries that represent the :class:`Composes <Compose>` we should run. 

219 

220 This method is compatible with the unversioned composer.start message, and also version 2. 

221 If no version is found, it will use the updates listed in the message to create new Compose 

222 objects and return dictionary representations of them. 

223 

224 This method also marks the Composes as pending, which acknowledges the receipt of the 

225 message. 

226 

227 Args: 

228 data: Information about the compose job we are processing. 

229 Returns: 

230 list: A list of dictionaries, as returned from :meth:`Compose.__json__`. 

231 """ 

232 with self.db_factory() as db: 

233 if api_version == 2: 

234 try: 

235 composes = [Compose.from_dict(db, c) for c in data['composes']] 

236 except sqlalchemy.orm.exc.NoResultFound: 

237 # It is possible for messages to get into our queue that reference Composes that 

238 # no longer exist. If this happens, we really just want to ignore the message so 

239 # that it gets dropped. In particular, we do not want to raise an Exception when 

240 # this happens, because that will Nack the message put it back into the queue, 

241 # resulting in a Nack loop. 

242 # See https://github.com/fedora-infra/bodhi/issues/3318 

243 log.info('Ignoring a compose task that references non-existing Composes') 

244 return [] 

245 else: 

246 raise ValueError('Unable to process request: {}'.format(data)) 

247 

248 # Filter out composes that are pending or have started, for example in 

249 # case of duplicate messages. 

250 composes = [c for c in composes if c.state == ComposeState.requested] 

251 

252 for c in composes: 

253 # Acknowledge that we've received the command to run these composes. 

254 c.state = ComposeState.pending 

255 

256 return [c.__json__(composer=True) for c in composes]

312 def run(self): 

313 """Run the thread by managing a db transaction and calling work().""" 

314 log.info('Grabbing semaphore') 

315 self.max_concur_sem.acquire() 

316 log.info('Acquired semaphore, starting') 

317 try: 

318 with self.db_factory() as session: 

319 self.db = session 

320 self.compose = Compose.from_dict(session, self._compose) 

321 self._checkpoints = json.loads(self.compose.checkpoints) 

322 log.info('Starting composer type %s for %s with %d updates', 

323 self, str(self.compose), len(self.compose.updates)) 

324 self.save_state(ComposeState.initializing) 

325 self.work() 

326 except Exception as e: 

327 with self.db_factory() as session: 

328 self.db = session 

329 self.compose = Compose.from_dict(session, self._compose) 

330 self.compose.error_message = str(e) 

331 self.save_state(ComposeState.failed) 

332 

333 log.exception('ComposerThread failed. Transaction rolled back.') 

334 finally: 

335 self.compose = None 

336 self.db = None 

337 self.max_concur_sem.release() 

338 log.info('Released semaphore')

352 def work(self): 

353 """Perform the various high-level tasks for the compose.""" 

354 self.id = getattr(self.compose.release, '%s_tag' % self.compose.request.value) 

355 

356 # Set our thread's "name" so it shows up nicely in the logs. 

357 # https://docs.python.org/2/library/threading.html#thread-objects 

358 self.name = self.id 

359 

360 # For 'pending' branched releases, we only want to perform repo-related 

361 # tasks for testing updates. For stable updates, we should just add the 

362 # dist_tag and do everything else other than composing/updateinfo, since 

363 # the nightly build-branched cron job composes for us. 

364 self.skip_compose = False 

365 if self.compose.release.state is ReleaseState.pending \ 

366 and self.compose.request is UpdateRequest.stable: 

367 self.skip_compose = True 

368 

369 log.info('Running ComposerThread(%s)' % self.id) 

370 

371 notifications.publish(compose_schemas.ComposeComposingV1.from_dict( 

372 dict(repo=self.id, 

373 updates=[' '.join([b.nvr for b in u.builds]) for u in self.compose.updates], 

374 agent=self.agent, 

375 ctype=self.ctype.value)), 

376 force=True, 

377 ) 

378 

379 try: 

380 if self.resume: 

381 self.load_state() 

382 else: 

383 self.save_state() 

384 

385 if self.compose.request is UpdateRequest.stable: 

386 self.perform_gating() 

387 

388 self.determine_and_perform_tag_actions() 

389 

390 self.update_security_bugs() 

391 

392 self.expire_buildroot_overrides() 

393 self.remove_pending_tags() 

394 

395 self._compose_updates() 

396 

397 self._mark_status_changes() 

398 self.save_state(ComposeState.notifying) 

399 self.send_notifications() 

400 

401 # Update bugzillas 

402 self.modify_bugs() 

403 

404 # Add comments to updates 

405 self.status_comments() 

406 

407 # Announce stable updates to the mailing list 

408 self.send_stable_announcements() 

409 

410 # Email updates-testing digest 

411 self.send_testing_digest() 

412 

413 self._unlock_updates() 

414 

415 self.check_all_karma_thresholds() 

416 self.obsolete_older_updates() 

417 

418 if config['clean_old_composes']: 

419 # Clean old composes 

420 self.save_state(ComposeState.cleaning) 

421 clean_old_composes(self.keep_old_composes) 

422 

423 self.save_state(ComposeState.success) 

424 self.success = True 

425 

426 self.remove_state() 

427 

428 except Exception: 

429 log.exception('Exception in ComposerThread(%s)' % self.id) 

430 self.save_state() 

431 raise 

432 finally: 

433 self.finish(self.success)

551 def _determine_tag_actions(self): 

552 tag_types, tag_rels = Release.get_tags(self.db) 

553 # sync & async tagging batches 

554 for i, batch in enumerate(sorted_updates(self.compose.updates)): 

555 for update in batch: 

556 add_tags = [] 

557 move_tags = [] 

558 

559 if update.status is UpdateStatus.testing: 

560 status = 'testing' 

561 else: 

562 status = 'candidate' 

563 

564 for build in update.builds: 

565 from_tag = None 

566 tags = build.get_tags() 

567 for tag in tags: 

568 if tag in tag_types[status]: 

569 from_tag = tag 

570 break 

571 else: 

572 reason = 'Cannot find relevant tag for %s. None of %s are in %s.' 

573 reason = reason % (build.nvr, tags, tag_types[status]) 

574 self.eject_from_compose(update, reason) 

575 break 

576 

577 if self.skip_compose: 

578 add_tags.append((update.requested_tag, build.nvr)) 

579 else: 

580 move_tags.append((from_tag, update.requested_tag, 

581 build.nvr)) 

582 else: 

583 if i == 0: 

584 self.add_tags_sync.extend(add_tags) 

585 self.move_tags_sync.extend(move_tags) 

586 else: 

587 self.add_tags_async.extend(add_tags) 

588 self.move_tags_async.extend(move_tags)

590 def _perform_tag_actions(self): 

591 koji = buildsys.get_session() 

592 for i, batches in enumerate([(self.add_tags_sync, self.move_tags_sync), 

593 (self.add_tags_async, self.move_tags_async)]): 

594 add, move = batches 

595 if i == 0: 

596 koji.multicall = False 

597 else: 

598 koji.multicall = True 

599 for action in add: 

600 tag, build = action 

601 log.info("Adding tag %s to %s" % (tag, build)) 

602 koji.tagBuild(tag, build, force=True) 

603 for action in move: 

604 from_tag, to_tag, build = action 

605 log.info('Moving %s from %s to %s' % (build, from_tag, to_tag)) 

606 koji.moveBuild(from_tag, to_tag, build, force=True) 

607 

608 if i != 0: 

609 results = koji.multiCall() 

610 failed_tasks = buildsys.wait_for_tasks([task[0] for task in results], 

611 koji, sleep=15) 

612 if failed_tasks: 

613 raise Exception("Failed to move builds: %s" % failed_tasks)

1018 def _get_master_repomd_url(self, arch): 

1019 """ 

1020 Return the master repomd URL for the given arch. 

1021 

1022 Look up the correct *_master_repomd setting in the config and use it to form the URL that 

1023 _wait_for_sync() will use to determine when the repository has been synchronized to the 

1024 master mirror. 

1025 

1026 Args: 

1027 arch (str): The architecture for which a URL needs to be formed. 

1028 

1029 Returns: 

1030 str: A URL on the master mirror where the repomd.xml file should be synchronized. 

1031 """ 

1032 release = self.compose.release.id_prefix.lower().replace('-', '_') 

1033 version = self.compose.release.version 

1034 request = self.compose.request.value 

1035 

1036 # First check to see if there's an override for the current version, if not, fall back. 

1037 # This will first try fedora_28_stable_(suffix), and then fedora_stable_(suffix). 

1038 key_prefixes = ['%s_%s_%s' % (release, version, request), 

1039 '%s_%s' % (release, request)] 

1040 # If the release has primary_arches defined in the config, we need to consider whether to 

1041 # use the release's *alt_master_repomd setting. 

1042 primary_arches = config.get( 

1043 '{release}_{version}_primary_arches'.format( 

1044 release=release, version=self.compose.release.version)) 

1045 if primary_arches and arch not in primary_arches.split(): 

1046 suffix = '_alt_master_repomd' 

1047 else: 

1048 suffix = '_master_repomd' 

1049 

1050 keys = [key_prefix + suffix for key_prefix in key_prefixes] 

1051 

1052 for key in keys: 

1053 val = config.get(key) 

1054 if val: 

1055 return val % (version, arch) 

1056 raise ValueError("Could not find any of %s in the config file" % ','.join(keys))

1058 def _punge(self): 

1059 """ 

1060 Launch the Pungi child process to "punge" the repository. 

1061 

1062 Returns: 

1063 subprocess.Popen: A process handle to the child Pungi process. 

1064 Raises: 

1065 Exception: If the child Pungi process exited with a non-0 exit code within 3 seconds. 

1066 """ 

1067 if self.path: 

1068 log.info('Skipping completed repo: %s', self.path) 

1069 return 

1070 

1071 # We have a thread-local devnull FD so that we can close them after the compose is done 

1072 self.devnull = open(os.devnull, 'wb') 

1073 

1074 self._create_pungi_config() 

1075 config_file = os.path.join(self._pungi_conf_dir, 'pungi.conf') 

1076 self._label = '%s-%s' % (config.get('pungi.labeltype'), 

1077 datetime.utcnow().strftime('%Y%m%d.%H%M')) 

1078 pungi_cmd = [config.get('pungi.cmd'), 

1079 '--config', config_file, 

1080 '--quiet', 

1081 '--print-output-dir', 

1082 '--target-dir', self.compose_dir, 

1083 '--old-composes', self.compose_dir, 

1084 '--no-latest-link', 

1085 '--label', self._label] 

1086 pungi_cmd += config.get('pungi.extracmdline') 

1087 

1088 log.info('Running the pungi command: %s', pungi_cmd) 

1089 compose_process = subprocess.Popen(pungi_cmd, 

1090 # Nope. No shell for you 

1091 shell=False, 

1092 # Should be useless, but just to set something 

1093 # predictable 

1094 cwd=self.compose_dir, 

1095 # Pungi will log the output compose dir to stdout 

1096 stdout=subprocess.PIPE, 

1097 # Stderr should also go to pungi.global.log if it starts 

1098 stderr=subprocess.PIPE, 

1099 # We will never have additional input 

1100 stdin=self.devnull) 

1101 log.info('Pungi running as PID: %s', compose_process.pid) 

1102 # Since the compose process takes a long time, we can safely just wait 3 seconds 

1103 # to abort the entire compose early if Pungi fails to start up correctly. 

1104 time.sleep(3) 

1105 if compose_process.poll() not in [0, None]: 

1106 log.error('Pungi process terminated with error within 3 seconds! Abandoning!') 

1107 _, err = compose_process.communicate() 

1108 log.error('Stderr: %s', err) 

1109 self.devnull.close() 

1110 raise Exception('Pungi returned error, aborting!') 

1111 

1112 return compose_process

1122 def _sanity_check_repo(self): 

1123 """Sanity check our repo. 

1124 

1125 - make sure we didn't compose a repo full of symlinks 

1126 - sanity check our repodata 

1127 

1128 This basically checks that pungi was run with gather_method='hardlink-or-copy' so that 

1129 we get a repository with either hardlinks or copied files. 

1130 This means that we when we go and sync generated repositories out, we do not need to take 

1131 special case to copy the target files rather than symlinks. 

1132 """ 

1133 log.info("Running sanity checks on %s" % self.path) 

1134 

1135 try: 

1136 arches = os.listdir(os.path.join(self.path, 'compose', 'Everything')) 

1137 except Exception: 

1138 log.exception('Empty compose folder? Compose thrown out') 

1139 self._toss_out_repo() 

1140 raise 

1141 

1142 if len(arches) == 0: 

1143 log.error('Empty compose, compose thrown out') 

1144 self._toss_out_repo() 

1145 raise Exception('Empty compose found') 

1146 

1147 for arch in arches: 

1148 # sanity check our repodata 

1149 try: 

1150 if arch == 'source': 

1151 repodata = os.path.join(self.path, 'compose', 

1152 'Everything', arch, 'tree', 'repodata') 

1153 sanity_check_repodata(repodata, repo_type='source') 

1154 else: 

1155 repodata = os.path.join(self.path, 'compose', 

1156 'Everything', arch, 'os', 'repodata') 

1157 repo_type = 'module' if self.ctype == ContentType.module else 'yum' 

1158 sanity_check_repodata(repodata, repo_type=repo_type) 

1159 except Exception: 

1160 log.exception("Repodata sanity check failed, compose thrown out") 

1161 self._toss_out_repo() 

1162 raise 

1163 

1164 # make sure that pungi didn't symlink our packages 

1165 try: 

1166 if arch == 'source': 

1167 dirs = [('tree', 'Packages')] 

1168 else: 

1169 dirs = [('debug', 'tree', 'Packages'), ('os', 'Packages')] 

1170 

1171 # Example of full path we are checking: 

1172 # self.path/compose/Everything/os/Packages/s/something.rpm 

1173 for checkdir in dirs: 

1174 checkdir = os.path.join(self.path, 'compose', 'Everything', arch, *checkdir) 

1175 subdirs = os.listdir(checkdir) 

1176 # subdirs is the self.path/compose/Everything/os/Packages/{a,b,c,...}/ dirs 

1177 # 

1178 # Let's check the first file in each subdir. If they are correct, we'll assume 

1179 # the rest is correct 

1180 # This is to avoid tons and tons of IOPS for a bunch of files put in in the 

1181 # same way 

1182 for subdir in subdirs: 

1183 for checkfile in os.listdir(os.path.join(checkdir, subdir)): 

1184 if not checkfile.endswith('.rpm'): 

1185 continue 

1186 if os.path.islink(os.path.join(checkdir, subdir, checkfile)): 

1187 log.error('Pungi out directory contains at least one ' 

1188 'symlink at %s', checkfile) 

1189 raise Exception('Symlinks found') 

1190 # We have checked the first rpm in the subdir 

1191 break 

1192 except Exception: 

1193 log.exception('Unable to check pungi composed repositories, compose thrown out') 

1194 self._toss_out_repo() 

1195 raise 

1196 

1197 return True

1211 def _wait_for_pungi(self, pungi_process): 

1212 """ 

1213 Wait for the pungi process to exit and find the path of the repository that it produced. 

1214 

1215 Args: 

1216 pungi_process (subprocess.Popen): The Popen handle of the running child process. 

1217 Raises: 

1218 Exception: If pungi's exit code is not 0, or if it is unable to find the directory that 

1219 Pungi created. 

1220 """ 

1221 self.save_state(ComposeState.punging) 

1222 if pungi_process is None: 

1223 log.info('Not waiting for pungi process, as there was no pungi') 

1224 return 

1225 log.info('Waiting for pungi process to finish') 

1226 out, err = pungi_process.communicate() 

1227 out = out.decode() 

1228 err = err.decode() 

1229 self.devnull.close() 

1230 if pungi_process.returncode != 0: 

1231 log.error('Pungi exited with exit code %d', pungi_process.returncode) 

1232 log.error('Stderr: %s', err) 

1233 raise Exception('Pungi exited with status %d' % pungi_process.returncode) 

1234 else: 

1235 log.info('Pungi finished') 

1236 

1237 # Find the path Pungi just created 

1238 prefix = 'Compose dir: ' 

1239 for line in out.split('\n'): 

1240 if line.startswith(prefix): 

1241 self.path = line[len(prefix):] 

1242 if not self.path:

1285 def _wait_for_sync(self): 

1286 """ 

1287 Block until our repomd.xml hits the master mirror. 

1288 

1289 Raises: 

1290 Exception: If no folder other than "source" was found in the compose_path. 

1291 """ 

1292 log.info('Waiting for updates to hit the master mirror') 

1293 notifications.publish(compose_schemas.ComposeSyncWaitV1.from_dict( 

1294 dict(repo=self.id, agent=self.agent)), 

1295 force=True) 

1296 compose_path = os.path.join(self.path, 'compose', 'Everything') 

1297 checkarch = None 

1298 # Find the first non-source arch to check against 

1299 for arch in os.listdir(compose_path): 

1300 if arch == 'source': 

1301 continue 

1302 checkarch = arch 

1303 break 

1304 if not checkarch: 

1305 raise Exception('Not found an arch to _wait_for_sync with') 

1306 

1307 repomd = os.path.join(compose_path, arch, 'os', 'repodata', 'repomd.xml') 

1308 if not os.path.exists(repomd): 

1309 log.error('Cannot find local repomd: %s', repomd) 

1310 return 

1311 

1312 self.save_state(ComposeState.syncing_repo) 

1313 master_repomd_url = self._get_master_repomd_url(arch) 

1314 

1315 with open(repomd) as repomdf: 

1316 checksum = hashlib.sha1(repomdf.read().encode('utf-8')).hexdigest() 

1317 while True: 

1318 try: 

1319 log.info('Polling %s' % master_repomd_url) 

1320 masterrepomd = urlopen(master_repomd_url) 

1321 newsum = hashlib.sha1(masterrepomd.read()).hexdigest() 

1322 except (ConnectionResetError, IncompleteRead, URLError, HTTPError): 

1323 log.exception('Error fetching repomd.xml') 

1324 time.sleep(200) 

1325 continue 

1326 if newsum == checksum: 

1327 log.info("master repomd.xml matches!") 

1328 notifications.publish(compose_schemas.ComposeSyncDoneV1.from_dict( 

1329 dict(repo=self.id, agent=self.agent)), 

1330 force=True) 

1331 return 

1332 

1333 log.debug("master repomd.xml doesn't match! %s != %s for %r", 

1334 checksum, newsum, self.id) 

1335 time.sleep(200)

1417 def _add_build_to_newest_builds(self, newest_builds, koji_build, override=False): 

1418 """ 

1419 Add Koji build to newest_builds dict if it's newer than the one there or override is set. 

1420

1428 not newer than the one currently stored there. 

1429 """ 

1430 # name:stream:version(.context) maps to Koji's name-version-release. 

1431 ns = "%s:%s" % (koji_build["name"], koji_build["version"]) 

1432 version = koji_build["release"] 

1433 context = '' 

1434 if '.' in version: 1434 ↛ 1437line 1434 didn't jump to line 1437, because the condition on line 1434 was never false

1435 version, context = version.split('.', 1)

1439 'version': version, 

1440 'context': context}

1449 def _generate_module_list(self): 

1450 """ 

1451 Generate a list of modules which should be used for pungi modular compose. 

1452 

1453 Returns: 

1454 list: list of moduledef dicts with name, stream, version and context 

1455 """ 

1456 # For modules, both name and version can contain dashes. This makes it 

1457 # impossible to distinguish between name and version from "nvr". We 

1458 # therefore have to ask for Koji build here and get that information 

1459 # from there. 

1460 koji = buildsys.get_session() 

1461 koji.multicall = True 

1462 for build in self.compose.release.builds: 

1463 koji.getBuild(build.nvr) 

1464 results = koji.multiCall() 

1465 

1466 # we loop through builds so we get rid of older builds and get only 

1467 # a dict with the newest builds 

1468 newest_builds = {} 

1469 for result, build in zip(results, self.compose.release.builds): 

1470 self._raise_on_get_build_multicall_error(result, build) 

1471 koji_build = result[0] 

1472 self._add_build_to_newest_builds(newest_builds, koji_build) 

1473 

1474 # make sure that the modules we want to update get their correct versions 

1475 for update in self.compose.updates: 

1476 # We need to get the Koji builds also for the updates. 

1477 koji.multicall = True 

1478 for update in self.compose.updates: 

1479 for build in update.builds: 

1480 koji.getBuild(build.nvr) 

1481 results = koji.multiCall() 

1482 for result, build in zip(results, update.builds): 

1483 self._raise_on_get_build_multicall_error(result, build) 

1484 koji_build = result[0] 

1485 self._add_build_to_newest_builds(newest_builds, koji_build, True) 

1486 

1487 # The keys are just used for easy name-stream finding. The name and stream are already in 

1488 # the module definitions. 

1489 return newest_builds.values()
