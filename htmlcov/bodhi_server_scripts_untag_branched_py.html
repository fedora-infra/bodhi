18""" 

19Used to remove the pending and testing tags from updates in a branched release. 

20 

21Since a separate task composes the branched stable repos, this will leave 

22those stable updates with the testing tags for 1 day before untagging. 

23 

24https://github.com/fedora-infra/bodhi/issues/576 

25""" 

26 

27from datetime import datetime, timedelta 

28import os 

29import sys 

30import logging 

31 

32from pyramid.paster import get_appsettings 

33 

34from bodhi.server import buildsys, Session, initialize_db 

35from bodhi.server.logging import setup as setup_logging 

36from ..models import Release, ReleaseState, Update, UpdateStatus 

37 

38 

39def usage(argv): 

40 """ 

41 Print usage information and exit with code 1. 

42 

43 Args: 

44 argv (list): The arguments passed to the script. 

45 """ 

46 cmd = os.path.basename(argv[0]) 

47 print('usage: %s <config_uri>\n' 

48 '(example: "%s development.ini")' % (cmd, cmd)) 

49 sys.exit(1) 

50 

51 

52def main(argv=sys.argv): 

53 """ 

54 Remove the pending and testing tags from branched updates. 

55 

56 Args: 

57 argv (list): The arguments passed to the script. Defaults to sys.argv. 

58 """ 

59 if len(argv) != 2: 

60 usage(argv) 

61 

62 config_uri = argv[1] 

63 

64 setup_logging() 

65 log = logging.getLogger(__name__) 

66 

67 settings = get_appsettings(config_uri) 

68 initialize_db(settings) 

69 db = Session() 

70 koji = buildsys.get_session() 

71 one_day = timedelta(days=1) 

72 now = datetime.utcnow() 

73 

74 try: 

75 for release in db.query(Release).filter_by( 

76 state=ReleaseState.pending).all(): 

77 log.info(release.name) 

78 for update in db.query(Update).filter_by( 

79 release=release, status=UpdateStatus.stable).all(): 

80 if now - update.date_stable > one_day: 

81 for build in update.builds: 

82 tags = build.get_tags() 

83 stable_tag = release.dist_tag 

84 testing_tag = release.testing_tag 

85 pending_signing_tag = release.pending_signing_tag 

86 pending_testing_tag = release.pending_testing_tag 

87 if stable_tag not in tags: 

88 log.error('%s not tagged as stable %s' % (build.nvr, tags)) 

89 continue 

90 if testing_tag in tags: 

91 log.info('Removing %s from %s' % (testing_tag, build.nvr)) 

92 koji.untagBuild(testing_tag, build.nvr) 

93 if pending_signing_tag in tags: 

94 log.info('Removing %s from %s' % (pending_signing_tag, build.nvr)) 

95 koji.untagBuild(pending_signing_tag, build.nvr) 

96 if pending_testing_tag in tags: 

97 log.info('Removing %s from %s' % (pending_testing_tag, build.nvr)) 

98 koji.untagBuild(pending_testing_tag, build.nvr) 

99 db.commit() 

100 except Exception as e: 

101 log.error(e) 

102 db.rollback() 

103 Session.remove() 

104 sys.exit(1)
