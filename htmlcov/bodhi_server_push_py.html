35def update_sig_status(update): 

36 """Update build signature status for builds in update.""" 

37 global _koji 

38 if _koji is None: 

39 # We don't want to authenticate to the buildsystem, because this script is often mistakenly 

40 # run as root and this can cause the ticket cache to become root owned with 0600 perms, 

41 # which will cause the compose to fail when it tries to use it to authenticate to Koji. 

42 buildsys.setup_buildsystem(config, authenticate=False) 

43 _koji = get_koji(None) 

44 for build in update.builds: 

45 if not build.signed: 

46 build_tags = build.get_tags(_koji) 

47 if update.release.pending_signing_tag not in build_tags: 

48 click.echo('Build %s was refreshed as signed' % build.nvr) 

49 build.signed = True 

50 else: 

51 click.echo('Build %s still unsigned' % build.nvr)

54def check_if_updates_and_builds_set( 

55 ctx: click.core.Context, param: click.core.Option, value: str) -> str: 

56 """ 

57 Print an error to stderr if the user has set both the --updates and --builds flags. 

58 

59 Args: 

60 ctx: The Click context, used to find out if the other flags are set. 

61 param: The option being handled. 

62 value: The value of the param flag. 

63 Returns: 

64 The value of the param flag. 

65 """ 

66 if value is not None and ((param.name == 'builds' and ctx.params.get('updates', False)) 

67 or (param.name == 'updates' and ctx.params.get('builds', False))): 

68 click.echo('ERROR: Must specify only one of --updates or --builds', err=True) 

69 sys.exit(1) 

70 return value 

71 

72 

73@click.command() 

74@click.option('--builds', help='Push updates for a comma-separated list of builds', 

75 callback=check_if_updates_and_builds_set) 

76@click.option('--updates', help='Push updates for a comma-separated list of update aliases', 

77 callback=check_if_updates_and_builds_set) 

78@click.option('--releases', help=('Push updates for a comma-separated list of releases (default: ' 

79 'current and pending releases)')) 

80@click.option('--request', default='testing,stable', 

81 help='Push updates with a specific request (default: testing,stable)') 

82@click.option('--resume', help='Resume one or more previously failed pushes', 

83 is_flag=True, default=False) 

84@click.option('--username', prompt=True) 

85@click.option('--yes', '-y', is_flag=True, default=False, 

86 help='Answers yes to the various questions') 

87@click.version_option(message='%(version)s') 

88def push(username, yes, **kwargs): 

89 """Push builds out to the repositories.""" 

90 resume = kwargs.pop('resume') 

91 resume_all = False 

92 

93 initialize_db(config) 

94 db_factory = transactional_session_maker() 

95 

96 composes = [] 

97 with db_factory() as session: 

98 if not resume and session.query(Compose).count(): 

99 if yes: 

100 click.echo('Existing composes detected: {}. Resuming all.'.format( 

101 ', '.join([str(c) for c in session.query(Compose).all()]))) 

102 else: 

103 click.confirm( 

104 'Existing composes detected: {}. Do you wish to resume them all?'.format( 

105 ', '.join([str(c) for c in session.query(Compose).all()])), 

106 abort=True) 

107 resume = True 

108 resume_all = True 

109 

110 # If we're resuming a push 

111 if resume: 

112 for compose in session.query(Compose).all(): 

113 if len(compose.updates) == 0: 

114 # Compose objects can end up with 0 updates in them if the composer ejects all 

115 # the updates in a compose for some reason. Composes with no updates cannot be 

116 # serialized because their content_type property uses the content_type of the 

117 # first update in the Compose. Additionally, it doesn't really make sense to go 

118 # forward with running an empty Compose. It makes the most sense to delete them. 

119 click.echo("{} has no updates. It is being removed.".format(compose)) 

120 session.delete(compose) 

121 continue 

122 

123 if not resume_all: 

124 if yes: 

125 click.echo('Resuming {}.'.format(compose)) 

126 elif not click.confirm('Resume {}?'.format(compose)): 

127 continue 

128 

129 # Reset the Compose's state and error message. 

130 compose.state = ComposeState.requested 

131 compose.error_message = '' 

132 

133 composes.append(compose) 

134 else: 

135 updates = [] 

136 # Accept both comma and space separated request list 

137 requests = kwargs['request'].replace(',', ' ').split(' ') 

138 requests = [UpdateRequest.from_string(val) for val in requests] 

139 

140 query = session.query(Update).filter(Update.request.in_(requests)) 

141 

142 if kwargs.get('builds'): 

143 query = query.join(Update.builds) 

144 query = query.filter( 

145 or_(*[Build.nvr == build for build in kwargs['builds'].split(',')])) 

146 

147 if kwargs.get('updates'): 

148 query = query.filter( 

149 or_(*[Update.alias == alias for alias in kwargs['updates'].split(',')])) 

150 

151 query = _filter_releases(session, query, kwargs.get('releases')) 

152 

153 for update in query.all(): 

154 # Skip unsigned updates (this checks that all builds in the update are signed) 

155 update_sig_status(update) 

156 

157 if not update.signed: 

158 click.echo( 

159 f'Warning: {update.get_title()} has unsigned builds and has been skipped', 

160 err=True) 

161 continue 

162 

163 updates.append(update) 

164 

165 composes = Compose.from_updates(updates) 

166 for c in composes: 

167 session.add(c) 

168 

169 # We need to flush so the database knows about the new Compose objects, so the 

170 # Compose.updates relationship will work properly. This is due to us overriding the 

171 # primaryjoin on the relationship between Composes and Updates. 

172 session.flush() 

173 

174 # Now we need to refresh the composes so their updates property will not be empty. 

175 for compose in composes: 

176 session.refresh(compose) 

177 

178 # Now we need to sort the composes so their security property can be used to prioritize 

179 # security updates. The security property relies on the updates property being 

180 # non-empty, so this must happen after the refresh above. 

181 composes = sorted(composes) 

182 

183 for compose in composes: 

184 click.echo('\n\n===== {} =====\n'.format(compose)) 

185 for update in compose.updates: 

186 click.echo(update.get_title()) 

187 

188 if composes: 

189 if yes: 

190 click.echo('\n\nPushing {:d} updates.'.format( 

191 sum([len(c.updates) for c in composes]))) 

192 else: 

193 click.confirm('\n\nPush these {:d} updates?'.format( 

194 sum([len(c.updates) for c in composes])), abort=True) 

195 click.echo('\nLocking updates...') 

196 else: 

197 click.echo('\nThere are no updates to push.') 

198 

199 composes = [c.__json__(composer=True) for c in composes] 

200 

201 if composes: 

202 click.echo('\nRequesting a compose') 

203 compose_task.delay(api_version=2, composes=composes, resume=resume, agent=username)

206def _filter_releases(session, query, releases=None): 

207 """ 

208 Filter the given query by releases. 

209 

210 Apply a filter() transformation to the given query on Updates to filter updates that match the 

211 given releases argument. If releases evaluates "Falsey", this function will filter for Updates 

212 that are part of a current Release. 

213 

214 :param session: The database session 

215 :param query: An Update query that we want to modify by filtering based on Releases 

216 :param releases: A comma-separated string of release names 

217 

218 :returns: A filtered version of query with an additional filter based on releases. 

219 """ 

220 # We will store models.Release object here that we want to filter by 

221 _releases = [] 

222 

223 # Filter only releases composed by Bodhi. 

224 releases_query = session.query(Release).filter(Release.composed_by_bodhi == True) 

225 

226 # Filter only releases that are current or pending. 

227 releases_query = releases_query.filter(or_(Release.state == ReleaseState.current, 

228 Release.state == ReleaseState.pending)) 

229 

230 if releases: 

231 for r in releases.split(','): 

232 release = releases_query.filter( 

233 or_(Release.name == r, 

234 Release.name == r.upper(), 

235 Release.version == r)).first() 

236 if not release: 

237 raise click.BadParameter( 

238 'Unknown release, or release not allowed to be composed: %s' % r 

239 ) 

240 else: 

241 _releases.append(release) 

242 else: 

243 # Since the user didn't ask for specific Releases, let's just filter for releases that are 

244 # current or pending. 

245 _releases = releases_query 

246 

247 return query.filter(or_(*[Update.release == r for r in _releases]))
