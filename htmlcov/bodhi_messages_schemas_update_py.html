30class UpdateMessage(BodhiMessage): 

31 """Base class for update messages.""" 

32 

33 @property 

34 def url(self) -> str: 

35 """ 

36 Return a URL to the action that caused this message to be emitted. 

37 

38 Returns: 

39 A relevant URL. 

40 """ 

41 return f"https://bodhi.fedoraproject.org/updates/{self.update.alias}" 

42 

43 @property 

44 def update(self) -> UpdateV1: 

45 """Return the Update referenced by this message.""" 

46 # Many things use this object, so let's cache it so we don't construct it repeatedly. 

47 if not hasattr(self, '_update_obj'): 

48 self._update_obj = UpdateV1( 

49 self._update['alias'], [BuildV1(b['nvr']) for b in self._update['builds']], 

50 UserV1(self._update['user']['name']), self._update['status'], 

51 self._update['request'], ReleaseV1(self._update['release']['name'])) 

52 return self._update_obj 

53 

54 @property 

55 def usernames(self) -> typing.List[str]: 

56 """ 

57 List of users affected by the action that generated this message. 

58 

59 Returns: 

60 A list of affected usernames. 

61 """ 

62 usernames = super(UpdateMessage, self).usernames 

63 # Add the submitter if there is one 

64 if self.update.user.name not in usernames: 

65 usernames.append(self.update.user.name) 

66 usernames.sort() 

67 return usernames 

68 

69 @property 

70 def packages(self) -> typing.Iterable[str]: 

71 """ 

72 List of names of packages affected by the action that generated this message. 

73 

74 Returns: 

75 A list of affected package names. 

76 """ 

77 return self.update.packages 

78 

79 @property 

80 def _update(self) -> dict: 

81 """Return a dictionary from the body representing an update.""" 

82 return self.body['update']

397class UpdateRequestMessage(UpdateMessage): 

398 """Sent when an update's request is changed.""" 

399 

400 @property 

401 def summary(self) -> str: 

402 """ 

403 Return a short, human-readable representation of this message. 

404 

405 This should provide a short summary of the message, much like the subject line 

406 of an email. 

407 

408 Returns: 

409 A summary for this message. 

410 """ 

411 status = self.topic.split('.')[-1] 

412 if status in ('unpush', 'obsolete', 'revoke'): 

413 # make our status past-tense 

414 status = status + (status[-1] == 'e' and 'd' or 'ed') 

415 return f"{self.agent} {status} {self.update.alias}" 

416 else: 

417 return f"{self.agent} submitted {self.update.alias} to {status}"
