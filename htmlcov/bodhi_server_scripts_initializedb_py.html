18"""Initialize the database on a new Bodhi installation.""" 

19 

20import os 

21import sys 

22 

23from pyramid.paster import get_appsettings 

24 

25from bodhi.server import initialize_db 

26from bodhi.server.logging import setup as setup_logging 

27from ..models import Base 

28 

29 

30def usage(argv): 

31 """ 

32 Print usage information and exit with code 1. 

33 

34 Args: 

35 argv (list): A list of command line arguments. 

36 """ 

37 cmd = os.path.basename(argv[0]) 

38 print('usage: %s <config_uri>\n' 

39 '(example: "%s development.ini")' % (cmd, cmd)) 

40 sys.exit(1) 

41 

42 

43def main(argv=sys.argv): 

44 """ 

45 Initialize the Bodhi database on a new installation. 

46 

47 This function is the entry point used by the initialize_bodhi_db console script. 

48 

49 Args: 

50 argv (list): A list of command line arguments. Defaults to sys.argv. 

51 """ 

52 if len(argv) != 2: 

53 usage(argv) 

54 config_uri = argv[1] 

55 setup_logging() 

56 settings = get_appsettings(config_uri) 

57 engine = initialize_db(settings) 

58 Base.metadata.bind = engine 

59 Base.metadata.create_all(engine)
