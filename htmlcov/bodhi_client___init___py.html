39def _warn_if_url_or_openid_and_staging_set( 

40 ctx: click.core.Context, param: typing.Union[click.core.Option, click.core.Parameter], 

41 value: str) -> str: 

42 """ 

43 Print a warning to stderr if the user has set both the --url/--openid-api and --staging flags. 

44 

45 This ensures that the user is aware that --staging supersedes --url/--openid-api. 

46 

47 Args: 

48 ctx: The Click context, used to find out if the --staging flag is set. 

49 param: The option being handled. 

50 value: The value of the option being handled. 

51 Returns: 

52 The value of the option being handled. 

53 """ 

54 if ctx.params.get('staging', False) and (param.name in ['url', 'openid_api']) and \ 

55 value is not None: 

56 click.echo(f'\nWarning: {param.name} and staging flags are ' 

57 f'both set. {param.name} will be ignored.\n', 

58 err=True) 

59 if param.name == 'staging' and value: 

60 if ctx.params.get('url', False): 

61 click.echo('\nWarning: url and staging flags are both set. url will be ignored.\n', 

62 err=True) 

63 if ctx.params.get('openid_api', False): 

64 click.echo('\nWarning: openid_api and staging flags ' 

65 'are both set. openid_api will be ignored.\n', 

66 err=True) 

67 return value

210def add_options(options: typing.Sequence[typing.Callable[[typing.Callable], typing.Callable]]) \ 

211 -> typing.Callable[[typing.Callable], typing.Callable]: 

212 """ 

213 Generate a click.option decorator with the given options. 

214 

215 Given a list of click options this creates a decorator that 

216 will return a function used to add the options to a click command. 

217 

218 Args: 

219 options: A list of click.options decorators. 

220 Returns: 

221 A decorator that applies the given options to it decorated function. 

222 """ 

223 def _add_options(func: typing.Callable) -> typing.Callable: 

224 """ 

225 Decorate func with the given click options. 

226 

227 Given a click command and a list of click options this will 

228 return the click command decorated with all the options in the list. 

229 

230 Args: 

231 func (callable): A click command function. 

232 Returns: 

233 callable: A wrapped version of func with added options. 

234 """ 

235 for option in reversed(options): 

236 func = option(func) 

237 return func 

238 return _add_options

241def handle_errors(method: typing.Callable) -> typing.Callable: 

242 """ 

243 Echo neat error messages on AuthError or BodhiClientException. 

244 

245 This is intended to be used as a decorator on method. 

246 

247 Args: 

248 method: The method we wish to handle errors from. 

249 Returns: 

250 A wrapped version of method that handles errors. 

251 """ 

252 @functools.wraps(method) 

253 def wrapper(*args, **kwargs): 

254 """ 

255 Call method with given args and kwargs, handling errors and exit if any are raised. 

256 

257 AuthErrors will cause an exit code of 1. BodhiClientExceptions will result in 2. 

258 Args: 

259 args: Positional arguments to pass to method. 

260 kwargs: Keyword arguments to pass to method. 

261 """ 

262 try: 

263 method(*args, **kwargs) 

264 except AuthError as e: 

265 click.secho(f"{e}: Check your FAS username & password", fg='red', bold=True, err=True) 

266 sys.exit(1) 

267 except bindings.BodhiClientException as e: 

268 click.secho(str(e), fg='red', bold=True, err=True) 

269 sys.exit(2) 

270 return wrapper

498

500 """

547 del kwargs['builds'] 

548 else: 

549 kwargs.pop('from_tag') 

550 if kwargs['addbuilds']: 

551 for build in kwargs['addbuilds'].split(','): 

552 if build not in kwargs['builds']: 552 ↛ 551line 552 didn't jump to line 551, because the condition on line 552 was never false

553 kwargs['builds'].append(build)

563 kwargs[field] = former_update[field]

565 require_severity_for_security_update(type=kwargs['type'], severity=kwargs['severity']) 

566

569 except bindings.BodhiClientException as e: 

570 click.echo(str(e), err=True)

742 help=('Include debuginfo packages')) 

743@click.option('--updateid', help='Download update(s) by ID(s) (comma-separated list)') 

744@click.option('--builds', help='Download update(s) by build NVR(s) (comma-separated list)') 

745@url_option 

746@debug_option 

747@handle_errors 

748def download(url: str, **kwargs): 

749 # User Docs that show in the --help 

750 """Download the builds in one or more updates.""" 

751 # Developer Docs 

752 """ 

753 Download the builds for an update. 

754 

755 Args: 

756 staging: Whether to use the staging server or not. 

757 arch: Requested architecture of packages to download. 

758 "all" will retrieve packages from all architectures. 

759 debuginfo: Whether to include debuginfo packages. 

760 url: The URL of a Bodhi server to create the update on. Ignored if staging is True. 

761 kwargs: Other keyword arguments passed to us by click. 

762 """ 

763 client = bindings.BodhiClient(base_url=url, staging=kwargs['staging']) 

764 requested_arch = kwargs['arch'] 

765 debuginfo = kwargs['debuginfo'] 

766 

767 del(kwargs['staging']) 

768 del(kwargs['arch']) 

769 del(kwargs['debuginfo']) 

770 # At this point we need to have reduced the kwargs dict to only our 

771 # query options (updateid or builds) 

772 if not any(kwargs.values()): 

773 click.echo("ERROR: must specify at least one of --updateid or --builds", err=True) 

774 sys.exit(1) 

775 

776 # As the query method doesn't let us construct OR queries, we're 

777 # gonna run one query for each option that was passed. The syntax 

778 # for this is a bit ugly, sorry. 

779 for (attr, value) in kwargs.items(): 

780 if value: 

781 expecteds = len(value.split(',')) 

782 resp = client.query(**{attr: value}) 

783 if len(resp.updates) == 0: 

784 click.echo(f"WARNING: No {attr} found!", err=True) 

785 else: 

786 if attr == 'updateid': 

787 resp_no = len(resp.updates) 

788 else: 

789 # attr == 'builds', add up number of builds for each returned update 

790 resp_no = functools.reduce( 

791 lambda x, y: x + len(y.get('builds', [])), resp.updates, 0 

792 ) 

793 

794 if resp_no < expecteds: 

795 click.echo(f"WARNING: Some {attr} not found!", err=True) 

796 # Not sure if we need a check for > expecteds, I don't 

797 # *think* that should ever be possible for these opts. 

798 

799 for update in resp.updates: 

800 click.echo(f"Downloading packages from {update['alias']}") 

801 for build in update['builds']: 

802 args = ['koji', 'download-build'] 

803 if debuginfo: 

804 args.append('--debuginfo') 

805 # subprocess is icky, but koji module doesn't 

806 # expose this in any usable way, and we don't want 

807 # to rewrite it here. 

808 if requested_arch is None: 

809 args.extend(['--arch=noarch', 

810 f'--arch={platform.machine()}', build['nvr']]) 

811 else: 

812 if 'all' in requested_arch: 

813 args.append(build['nvr']) 

814 if 'all' not in requested_arch: 

815 args.extend(['--arch=noarch', 

816 f'--arch={requested_arch}', build['nvr']]) 

817 ret = subprocess.call(args) 

818 if ret: 

819 click.echo(f"WARNING: download of {build['nvr']} failed!", err=True)

1113def print_resp(resp: munch.Munch, client: bindings.BodhiClient, verbose: bool = False, 

1114 override_hint: bool = True): 

1115 """ 

1116 Print a human readable rendering of the given server response to the terminal. 

1117 

1118 Args: 

1119 resp: The response from the server. 

1120 client: A BodhiClient. 

1121 verbose: If True, show more detailed output. Defaults to False. 

1122 override_hint: If True, show a hint to the user about how to wait on a buildroot 

1123 override. Defaults to True. 

1124 """ 

1125 if 'updates' in resp: 

1126 if len(resp.updates) == 1: 

1127 click.echo(client.update_str(resp.updates[0])) 

1128 else: 

1129 for update in resp.updates: 

1130 click.echo(client.update_str(update, minimal=True)) 

1131 if 'total' in resp: 

1132 click.echo(f'{resp.total} updates found ({len(resp.updates)} shown)') 

1133 elif resp.get('update'): 

1134 click.echo(client.update_str(resp['update'])) 

1135 elif resp.get('alias'): 

1136 click.echo(client.update_str(resp)) 

1137 elif 'overrides' in resp: 

1138 if len(resp.overrides) == 1: 

1139 click.echo(client.override_str(resp.overrides[0], minimal=False)) 

1140 if override_hint: 1140 ↛ 1145line 1140 didn't jump to line 1145, because the condition on line 1140 was never false

1141 _print_override_koji_hint(resp.overrides[0], client) 

1142 else: 

1143 for override in resp.overrides: 

1144 click.echo(client.override_str(override).strip()) 

1145 if 'total' in resp: 

1146 click.echo( 

1147 f'{resp.total} overrides found ({len(resp.overrides)} shown)') 

1148 elif 'build' in resp: 

1149 click.echo(client.override_str(resp, minimal=False)) 

1150 if override_hint: 

1151 _print_override_koji_hint(resp, client) 

1152 elif 'comment' in resp: 

1153 click.echo(f"The following comment was added to {resp.comment['update'].alias}") 

1154 click.echo(resp.comment.text) 

1155 elif 'compose' in resp: 

1156 click.echo(client.compose_str(resp['compose'], minimal=False)) 

1157 elif 'composes' in resp: 

1158 if len(resp['composes']) == 1: 

1159 click.echo(client.compose_str(resp['composes'][0], minimal=(not verbose))) 

1160 else: 

1161 for compose in resp['composes']: 

1162 click.echo(client.compose_str(compose, minimal=(not verbose))) 

1163 if verbose: 

1164 # Let's add a little more spacing 

1165 click.echo() 

1166 else: 

1167 click.echo(resp)
